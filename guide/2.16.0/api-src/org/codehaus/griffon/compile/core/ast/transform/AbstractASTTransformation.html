075     public static final ClassNode COLLECTIONS_CLASS = makeClassSafe(Collections.class);

087     @Nonnull
088     public static Expression applicationExpression(@Nonnull ClassNode classNode) {
089         FieldNode field = classNode.getDeclaredField(PROPERTY_APPLICATION);
090         if (field != null) {
091             return new FieldExpression(field);
092         }
093         field = classNode.getField(PROPERTY_APPLICATION);
094         if (field != null && !isPrivate(field.getModifiers())) {
095             return new FieldExpression(field);
096         }
097 
098         MethodNode method = classNode.getDeclaredMethod(METHOD_GET_APPLICATION, NO_PARAMS);
099         if (method != null) {
100             return call(THIS, METHOD_GET_APPLICATION, NO_ARGS);
101         }
102         method = classNode.getMethod(METHOD_GET_APPLICATION, NO_PARAMS);
103         if (method != null && !isPrivate(method.getModifiers())) {
104             return call(THIS, METHOD_GET_APPLICATION, NO_ARGS);
105         }
106         throw new IllegalStateException("Cannot resolve application field nor getApplication() method on class " + classNode.getName());
107     }

114     public static void injectApplication(@Nonnull ClassNode classNode) {
115         try {
116             applicationExpression(classNode);
117         } catch (IllegalStateException iae) {
118             FieldNode fieldNode = injectField(classNode, PROPERTY_APPLICATION, PRIVATE, GRIFFON_APPLICATION_TYPE, null, false);
119             List<AnnotationNode> annotations = fieldNode.getAnnotations(INJECT_TYPE);
120             if (annotations == null || annotations.isEmpty()) {
121                 fieldNode.addAnnotation(new AnnotationNode(INJECT_TYPE));
122             }
123         }
124 
125         MethodNode method = classNode.getDeclaredMethod(METHOD_GET_APPLICATION, NO_PARAMS);
126         if (method == null) {
127             injectMethod(classNode, new MethodNode(
128                 METHOD_GET_APPLICATION,
129                 PUBLIC,
130                 GRIFFON_APPLICATION_TYPE,
131                 NO_PARAMS,
132                 NO_EXCEPTIONS,
133                 returns(field(classNode, PROPERTY_APPLICATION))
134             ));
135         }
136         method = classNode.getMethod(METHOD_GET_APPLICATION, NO_PARAMS);
137         if (method == null) {
138             injectMethod(classNode, new MethodNode(
139                 METHOD_GET_APPLICATION,
140                 PUBLIC,
141                 GRIFFON_APPLICATION_TYPE,
142                 NO_PARAMS,
143                 NO_EXCEPTIONS,
144                 returns(field(classNode, PROPERTY_APPLICATION))
145             ));
146         }
147     }

153     public static FieldExpression injectedField(@Nonnull ClassNode owner, @Nonnull ClassNode type, @Nonnull String name, @Nullable String qualifierName) {
154         FieldNode fieldNode = GriffonASTUtils.injectField(owner, name, Modifier.PRIVATE, type, null, false);
155         List<AnnotationNode> annotations = fieldNode.getAnnotations(INJECT_TYPE);
156         if (annotations == null || annotations.isEmpty()) {
157             fieldNode.addAnnotation(new AnnotationNode(INJECT_TYPE));
158         }
159         if (isNotBlank(qualifierName)) {
160             AnnotationNode namedAnnotation = new AnnotationNode(NAMED_TYPE);
161             namedAnnotation.addMember("value", new ConstantExpression(qualifierName));
162             fieldNode.addAnnotation(namedAnnotation);
163         }
164         return new FieldExpression(fieldNode);
165     }

252     public static boolean needsDelegate(@Nonnull ClassNode classNode, @Nonnull SourceUnit sourceUnit,
253                                         @Nonnull MethodDescriptor[] methods, @Nonnull String annotationType,
254                                         @Nonnull String delegateType) {
255         boolean implemented = false;
256         int implementedCount = 0;
257         ClassNode consideredClass = classNode;
258         while (consideredClass != null) {
259             for (MethodNode method : consideredClass.getMethods()) {
260                 for (MethodDescriptor md : methods) {
261                     if (method.getName().equals(md.methodName) && method.getParameters().length == md.arguments.length) {
262                         implemented |= true;
263                         implementedCount++;
264                     }
265                     if (implementedCount == methods.length) {
266                         return false;
267                     }
268                 }
269             }
270 
271             consideredClass = consideredClass.getSuperClass();
272         }
273         if (implemented) {
274             sourceUnit.getErrorCollector().addErrorAndContinue(
275                 new SimpleMessage("@" + annotationType + " cannot be processed on "
276                     + classNode.getName()
277                     + " because some but not all methods from "
278                     + delegateType
279                     + " were declared in the current class or super classes.",
280                     sourceUnit)
281             );
282             return false;
283         }
284         return true;
285     }

287     public static void addDelegateMethods(@Nonnull ClassNode classNode, @Nonnull ClassNode delegateType, @Nonnull Expression delegate) {
288         for (MethodNode method : delegateType.getMethods()) {
289             List<Expression> variables = new ArrayList<>();
290             Parameter[] parameters = new Parameter[method.getParameters().length];
291             ClassNode[] exceptions = new ClassNode[method.getExceptions().length];
292             for (int i = 0; i < method.getParameters().length; i++) {
293                 Parameter p = method.getParameters()[i];
294                 parameters[i] = new Parameter(makeClassSafe(p.getType()), p.getName());
295                 parameters[i].getType().setGenericsTypes(p.getType().getGenericsTypes());
296                 parameters[i].getType().setGenericsPlaceHolder(p.getType().isGenericsPlaceHolder());
297                 parameters[i].getType().setUsingGenerics(p.getType().isUsingGenerics());
298                 parameters[i].addAnnotations(p.getAnnotations());
299                 variables.add(var(p.getName()));
300             }
301             for (int i = 0; i < method.getExceptions().length; i++) {
302                 ClassNode ex = method.getExceptions()[i];
303                 exceptions[i] = makeClassSafe(ex);
304             }
305             ClassNode returnType = makeClassSafe(method.getReturnType());
306             returnType.addAnnotations(method.getReturnType().getAnnotations());
307             returnType.setGenericsTypes(method.getReturnType().getGenericsTypes());
308             returnType.setGenericsPlaceHolder(method.getReturnType().isGenericsPlaceHolder());
309             returnType.setUsingGenerics(method.getReturnType().isUsingGenerics());
310 
311             boolean isVoid = ClassHelper.VOID_TYPE.equals(method.getReturnType());
312             Expression delegateExpression = call(
313                 delegate,
314                 method.getName(),
315                 args(variables));
316             MethodNode newMethod = new MethodNode(
317                 method.getName(),
318                 method.getModifiers() - Modifier.ABSTRACT,
319                 returnType,
320                 parameters,
321                 exceptions,
322                 isVoid ? stmnt(delegateExpression) : returns(delegateExpression)
323             );
324             newMethod.setGenericsTypes(method.getGenericsTypes());
325             injectMethod(classNode, newMethod);
326         }
327     }
