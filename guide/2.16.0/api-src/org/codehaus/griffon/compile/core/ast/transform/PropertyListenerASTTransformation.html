043 /**
044  * Handles generation of code for the {@code @PropertyListener} annotation.
045  * <p/>
046  * Any closures found as the annotation's value will be either transformed
047  * into inner classes that implement PropertyChangeListener (when the value
048  * is a closure defined in place) or be casted as a proxy of PropertyChangeListener
049  * (when the value is a property reference found in the same class).<p>
050  * List of closures are also supported.
051  *
052  * @author Andres Almiray
053  */
054 @GroovyASTTransformation(phase= CompilePhase.CANONICALIZATION)
055 public class PropertyListenerASTTransformation extends AbstractASTTransformation {
056     private static final ClassNode PROPERTY_LISTENER_CLASS = makeClassSafe(PropertyListener.class);
057     private static final ClassNode PROPERTY_CHANGE_LISTENER_CLASS = makeClassSafe(PropertyChangeListener.class);
058     private static final String EMPTY_STRING = "";
059 
060     /**
061      * Convenience method to see if an annotated node is {@code @PropertyListener}.
062      *
063      * @param node the node to check
064      * @return true if the node is an event publisher
065      */
066     public static boolean hasListenerAnnotation(AnnotatedNode node) {
067         for (AnnotationNode annotation : node.getAnnotations()) {
068             if (PROPERTY_LISTENER_CLASS.equals(annotation.getClassNode())) {
069                 return true;
070             }
071         }
072         return false;
073     }
074 
075     public static void addListenerToProperty(SourceUnit source, AnnotationNode annotation, ClassNode declaringClass, FieldNode field) {
076         for (Map.Entry<String, Expression> member : annotation.getMembers().entrySet()) {
077             Expression value = member.getValue();
078             if ((value instanceof ListExpression)) {
079                 for (Expression expr : ((ListExpression) value).getExpressions()) {
080                     processExpression(declaringClass, field.getName(), expr);
081                 }
082                 member.setValue(new ConstantExpression(EMPTY_STRING));
083             } else {
084                 processExpression(declaringClass, field.getName(), value);
085                 member.setValue(new ConstantExpression(EMPTY_STRING));
086             }
087         }
088     }
089 
090     public static void addListenerToClass(SourceUnit source, AnnotationNode annotation, ClassNode classNode) {
091         for (Map.Entry<String, Expression> member : annotation.getMembers().entrySet()) {
092             Expression value = member.getValue();
093             if ((value instanceof ListExpression)) {
094                 for (Expression expr : ((ListExpression) value).getExpressions()) {
095                     processExpression(classNode, null, expr);
096                 }
097                 member.setValue(new ConstantExpression(EMPTY_STRING));
098             } else {
099                 processExpression(classNode, null, value);
100                 member.setValue(new ConstantExpression(EMPTY_STRING));
101             }
102         }
103     }
104 
105     /**
106      * Handles the bulk of the processing, mostly delegating to other methods.
107      *
108      * @param nodes   the ast nodes
109      * @param source  the source unit for the nodes
110      */
111     public void visit(ASTNode[] nodes, SourceUnit source) {
112         if (!(nodes[0] instanceof AnnotationNode) || !(nodes[1] instanceof AnnotatedNode)) {
113             throw new RuntimeException("Internal error: wrong types: " + nodes[0].getClass() + " / " + nodes[1].getClass());
114         }
115 
116         AnnotationNode annotation = (AnnotationNode) nodes[0];
117         AnnotatedNode parent = (AnnotatedNode) nodes[1];
118 
119         ClassNode declaringClass = parent.getDeclaringClass();
120         if (parent instanceof FieldNode) {
121             addListenerToProperty(source, annotation, declaringClass, (FieldNode) parent);
122         } else if (parent instanceof ClassNode) {
123             addListenerToClass(source, annotation, (ClassNode) parent);
124         }
125     }
126 
127     private static void processExpression(ClassNode classNode, String propertyName, Expression expression) {
128         if (expression instanceof ClosureExpression) {
129             addPropertyChangeListener(classNode, propertyName, (ClosureExpression) expression);
130         } else if (expression instanceof VariableExpression) {
131             addPropertyChangeListener(classNode, propertyName, (VariableExpression) expression);
132         } else if (expression instanceof ConstantExpression) {
133             addPropertyChangeListener(classNode, propertyName, (ConstantExpression) expression);
134         } else {
135             throw new RuntimeException("Internal error: wrong expression type. " + expression);
136         }
137     }
138 
139     private static void addPropertyChangeListener(ClassNode classNode, String propertyName, ClosureExpression closure) {
140         ArgumentListExpression args = new ArgumentListExpression();
141         if (propertyName != null) args.addExpression(new ConstantExpression(propertyName));
142         args.addExpression(CastExpression.asExpression(PROPERTY_CHANGE_LISTENER_CLASS, closure));
143 
144         addListenerStatement(classNode, args);
145     }
146 
147     private static void addPropertyChangeListener(ClassNode classNode, String propertyName, VariableExpression variable) {
148         ArgumentListExpression args = new ArgumentListExpression();
149         if (propertyName != null) args.addExpression(new ConstantExpression(propertyName));
150         args.addExpression(CastExpression.asExpression(PROPERTY_CHANGE_LISTENER_CLASS, variable));
151 
152         addListenerStatement(classNode, args);
153     }
154 
155     private static void addPropertyChangeListener(ClassNode classNode, String propertyName, ConstantExpression reference) {
156         ArgumentListExpression args = new ArgumentListExpression();
157         if (propertyName != null) args.addExpression(new ConstantExpression(propertyName));
158         VariableExpression variable = new VariableExpression(reference.getText());
159         args.addExpression(CastExpression.asExpression(PROPERTY_CHANGE_LISTENER_CLASS, variable));
160 
161         addListenerStatement(classNode, args);
162     }
163 
164     private static void addListenerStatement(ClassNode classNode, ArgumentListExpression args) {
165         BlockStatement body = new BlockStatement();
166         body.addStatement(new ExpressionStatement(
167             new MethodCallExpression(
168                 VariableExpression.THIS_EXPRESSION,
169                 "addPropertyChangeListener",
170                 args
171             )
172         ));
173 
174         classNode.addObjectInitializerStatements(body);
175     }
