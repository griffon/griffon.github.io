33  /**
34   * <p> Utility reflection methods focussed on methods in general rather than properties in particular. </p>
35   * <p/>
36   * <p><b>Copied from commons-beanutils, removed dependencies to commons-logging</b></p>
37   * <p/>
38   * <h3>Known Limitations</h3>
39   * <h4>Accessing Public Methods In A Default Access Superclass</h4>
40   * <p>There is an issue when invoking public methods contained in a default access superclass.
41   * Reflection locates these methods fine and correctly assigns them as public.
42   * However, an <code>IllegalAccessException</code> is thrown if the method is invoked.</p>
43   * <p/>
44   * <p><code>MethodUtils</code> contains a workaround for this situation.
45   * It will attempt to call <code>setAccessible</code> on this method.
46   * If this call succeeds, then the method can be invoked as normal.
47   * This call will only succeed when the application has sufficient security privilages.
48   *
49   * @author Craig R. McClanahan
50   * @author Ralph Schaer
51   * @author Chris Audley
52   * @author Rey Fran&#231;ois
53   * @author Gregor Ra&#253;man
54   * @author Jan Sorensen
55   * @author Robert Burrell Donkin
56   */
57

66      /**
67       * Indicates whether methods should be cached for improved performance.
68       * <p/>
69       * Note that when this class is deployed via a shared classloader in
70       * a container, this will affect all webapps. However making this
71       * configurable per webapp would mean having a map keyed by context classloader
72       * which may introduce memory-leak problems.
73       */
74      private static boolean CACHE_METHODS = true;
75  
76      /**
77       * An empty class array
78       */
79      private static final Class[] EMPTY_CLASS_PARAMETERS = new Class[0];
80      /**
81       * An empty object array
82       */
83      private static final Object[] EMPTY_OBJECT_ARRAY = new Object[0];
84  
85      /**
86       * Stores a cache of MethodDescriptor -> Method in a WeakHashMap.
87       * <p/>
88       * The keys into this map only ever exist as temporary variables within
89       * methods of this class, and are never exposed to users of this class.
90       * This means that the WeakHashMap is used only as a mechanism for
91       * limiting the size of the cache, ie a way to tell the garbage collector
92       * that the contents of the cache can be completely garbage-collected
93       * whenever it needs the memory. Whether this is a good approach to
94       * this problem is doubtful; something like the commons-collections
95       * LRUMap may be more appropriate (though of course selecting an
96       * appropriate size is an issue).
97       * <p/>
98       * This static variable is safe even when this code is deployed via a
99       * shared classloader because it is keyed via a MethodDescriptor object
100      * which has a Class as one of its members and that member is used in
101      * the MethodDescriptor.equals method. So two components that load the same
102      * class via different classloaders will generate non-equal MethodDescriptor
103      * objects and hence end up with different entries in the map.
104      */
105     private static final Map cache = Collections.synchronizedMap(new WeakHashMap());

109     /**
110      * Set whether methods should be cached for greater performance or not,
111      * default is <code>true</code>.
112      *
113      * @param cacheMethods <code>true</code> if methods should be
114      *                     cached for greater performance, otherwise <code>false</code>
115      *
116      * @since 1.8.0
117      */
118     public static synchronized void setCacheMethods(boolean cacheMethods) {
119         CACHE_METHODS = cacheMethods;
120         if (!CACHE_METHODS) {
121             clearCache();
122         }
123     }

183      * <p/>
184      * <p>The behaviour of this method is less deterministic
185      * than {@link #invokeExactMethod(Object object, String methodName, Object [] args)}.

187      * and then executes the first it finds with compatable parameters.</p>
188      * <p/>
189      * <p>This method supports calls to methods taking primitive parameters
190      * via passing in wrapping classes. So, for example, a <code>Boolean</code> class
191      * would match a <code>boolean</code> primitive.</p>
192      * <p/>
193      * <p> This is a convenient wrapper for
194      * {@link #invokeMethod(Object object, String methodName, Object[] args, Class[] parameterTypes)}.
195      * </p>
196      *
197      * @param object     invoke method on this object
198      * @param methodName get method with this name
199      * @param args       use these arguments - treat null as empty array
200      *
201      * @return The value returned by the invoked method
202      *
203      * @throws NoSuchMethodException     if there is no such accessible method
204      * @throws InvocationTargetException wraps an exception thrown by the
205      *                                   method invoked
206      * @throws IllegalAccessException    if the requested method is not accessible
207      *                                   via reflection
208      */
209     public static Object invokeMethod(
210         Object object,
211         String methodName,
212         Object[] args)
213         throws
214         NoSuchMethodException,
215         IllegalAccessException,
216         InvocationTargetException {
217 
218         if (args == null) {
219             args = EMPTY_OBJECT_ARRAY;
220         }
221         int arguments = args.length;
222         Class[] parameterTypes = new Class[arguments];
223         for (int i = 0; i < arguments; i++) {
224             parameterTypes[i] = args[i] != null ? args[i].getClass() : Object.class;
225         }
226         return invokeMethod(object, methodName, args, parameterTypes);
227 
228     }

255      *                                   via reflection

257     public static Object invokeMethod(
258         Object object,
259         String methodName,
260         Object[] args,
261         Class[] parameterTypes)
262         throws
263         NoSuchMethodException,
264         IllegalAccessException,
265         InvocationTargetException {
266 
267         if (parameterTypes == null) {
268             parameterTypes = EMPTY_CLASS_PARAMETERS;
269         }
270         if (args == null) {
271             args = EMPTY_OBJECT_ARRAY;
272         }
273 
274         Method method = getMatchingAccessibleMethod(
275             object.getClass(),
276             methodName,
277             parameterTypes);
278         if (method == null) {
279             throw new NoSuchMethodException("No such accessible method: " +
280                 methodName + "() on object: " + object.getClass().getName());
281         }
282         return method.invoke(object, args);
283     }

773      * <p>Return an accessible method (that is, one that can be invoked via
774      * reflection) that implements the specified Method.  If no such method
775      * can be found, return <code>null</code>.</p>
776      *
777      * @param clazz  The class of the object
778      * @param method The method that we wish to call
779      *
780      * @return The accessible method
781      *
782      * @since 1.8.0
783      */
784     public static Method getAccessibleMethod(Class clazz, Method method) {
785 
786         // Make sure we have a method to check
787         if (method == null) {
788             return (null);
789         }
790 
791         // If the requested method is not public we cannot call it
792         if (!Modifier.isPublic(method.getModifiers())) {
793             return (null);
794         }
795 
796         boolean sameClass = true;
797         if (clazz == null) {
798             clazz = method.getDeclaringClass();
799         } else {
800             sameClass = clazz.equals(method.getDeclaringClass());
801             if (!method.getDeclaringClass().isAssignableFrom(clazz)) {
802                 throw new IllegalArgumentException(clazz.getName() +
803                     " is not assignable from " + method.getDeclaringClass().getName());
804             }
805         }
806 
807         // If the class is public, we are done
808         if (Modifier.isPublic(clazz.getModifiers())) {
809             if (!sameClass && !Modifier.isPublic(method.getDeclaringClass().getModifiers())) {
810                 setMethodAccessible(method); // Default access superclass workaround
811             }
812             return (method);
813         }
814 
815         String methodName = method.getName();
816         Class[] parameterTypes = method.getParameterTypes();
817 
818         // Check the implemented interfaces and subinterfaces
819         method =
820             getAccessibleMethodFromInterfaceNest(clazz,
821                 methodName,
822                 parameterTypes);
823 
824         // Check the superclass chain
825         if (method == null) {
826             method = getAccessibleMethodFromSuperclass(clazz,
827                 methodName,
828                 parameterTypes);
829         }
830 
831         return (method);
832     }

838      * <p>Return an accessible method (that is, one that can be invoked via
839      * reflection) by scanning through the superclasses. If no such method
840      * can be found, return <code>null</code>.</p>
841      *
842      * @param clazz          Class to be checked
843      * @param methodName     Method name of the method we wish to call
844      * @param parameterTypes The parameter type signatures
845      */
846     private static Method getAccessibleMethodFromSuperclass
847     (Class clazz, String methodName, Class[] parameterTypes) {
848 
849         Class parentClazz = clazz.getSuperclass();
850         while (parentClazz != null) {
851             if (Modifier.isPublic(parentClazz.getModifiers())) {
852                 try {
853                     return parentClazz.getMethod(methodName, parameterTypes);
854                 } catch (NoSuchMethodException e) {
855                     return null;
856                 }
857             }
858             parentClazz = parentClazz.getSuperclass();
859         }
860         return null;
861     }

863     /**
864      * <p>Return an accessible method (that is, one that can be invoked via
865      * reflection) that implements the specified method, by scanning through
866      * all implemented interfaces and subinterfaces.  If no such method
867      * can be found, return <code>null</code>.</p>
868      * <p/>
869      * <p> There isn't any good reason why this method must be private.
870      * It is because there doesn't seem any reason why other classes should
871      * call this rather than the higher level methods.</p>
872      *
873      * @param clazz          Parent class for the interfaces to be checked
874      * @param methodName     Method name of the method we wish to call
875      * @param parameterTypes The parameter type signatures
876      */
877     private static Method getAccessibleMethodFromInterfaceNest
878     (Class clazz, String methodName, Class[] parameterTypes) {
879 
880         Method method = null;
881 
882         // Search up the superclass chain
883         for (; clazz != null; clazz = clazz.getSuperclass()) {
884 
885             // Check the implemented interfaces of the parent class
886             Class[] interfaces = clazz.getInterfaces();
887             for (int i = 0; i < interfaces.length; i++) {
888 
889                 // Is this interface public?
890                 if (!Modifier.isPublic(interfaces[i].getModifiers())) {
891                     continue;
892                 }
893 
894                 // Does the method exist on this interface?
895                 try {
896                     method = interfaces[i].getDeclaredMethod(methodName,
897                         parameterTypes);
898                 } catch (NoSuchMethodException e) {
899                     /* Swallow, if no method is found after the loop then this
900                      * method returns null.
901                      */
902                 }
903                 if (method != null) {
904                     return method;
905                 }
906 
907                 // Recursively check our parent interfaces
908                 method =
909                     getAccessibleMethodFromInterfaceNest(interfaces[i],
910                         methodName,
911                         parameterTypes);
912                 if (method != null) {
913                     return method;
914                 }
915 
916             }
917 
918         }
919 
920         // We did not find anything
921         return (null);
922 
923     }

925     /**
926      * <p>Find an accessible method that matches the given name and has compatible parameters.
927      * Compatible parameters mean that every method parameter is assignable from
928      * the given parameters.
929      * In other words, it finds a method with the given name
930      * that will take the parameters given.<p>
931      * <p/>
932      * <p>This method is slightly undeterminstic since it loops
933      * through methods names and return the first matching method.</p>
934      * <p/>
935      * <p>This method is used by
936      * {@link
937      * #invokeMethod(Object object, String methodName, Object[] args, Class[] parameterTypes)}.
938      * <p/>
939      * <p>This method can match primitive parameter by passing in wrapper classes.
940      * For example, a <code>Boolean</code> will match a primitive <code>boolean</code>
941      * parameter.
942      *
943      * @param clazz          find method in this class
944      * @param methodName     find method with this name
945      * @param parameterTypes find method with compatible parameters
946      *
947      * @return The accessible method
948      */
949     public static Method getMatchingAccessibleMethod(
950         Class clazz,
951         String methodName,
952         Class[] parameterTypes) {
953         MethodDescriptordDescriptor">MethodDescriptor md = new MethodDescriptor(clazz, methodName, parameterTypes, false);
954 
955         // see if we can find the method directly
956         // most of the time this works and it's much faster
957         try {
958             // Check the cache first
959             Method method = getCachedMethod(md);
960             if (method != null) {
961                 return method;
962             }
963 
964             method = clazz.getMethod(methodName, parameterTypes);
965 
966             setMethodAccessible(method); // Default access superclass workaround
967 
968             cacheMethod(md, method);
969             return method;
970 
971         } catch (NoSuchMethodException e) { /* SWALLOW */ }
972 
973         // search through all methods 
974         int paramSize = parameterTypes.length;
975         Method bestMatch = null;
976         Method[] methods = clazz.getMethods();
977         float bestMatchCost = Float.MAX_VALUE;
978         float myCost = Float.MAX_VALUE;
979         for (int i = 0, size = methods.length; i < size; i++) {
980             if (methods[i].getName().equals(methodName)) {
981                 // compare parameters
982                 Class[] methodsParams = methods[i].getParameterTypes();
983                 int methodParamSize = methodsParams.length;
984                 if (methodParamSize == paramSize) {
985                     boolean match = true;
986                     for (int n = 0; n < methodParamSize; n++) {
987                         if (!isAssignmentCompatible(methodsParams[n], parameterTypes[n])) {
988                             match = false;
989                             break;
990                         }
991                     }
992 
993                     if (match) {
994                         // get accessible version of method
995                         Method method = getAccessibleMethod(clazz, methods[i]);
996                         if (method != null) {
997                             setMethodAccessible(method); // Default access superclass workaround
998                             myCost = getTotalTransformationCost(parameterTypes, method.getParameterTypes());
999                             if (myCost < bestMatchCost) {
1000                                 bestMatch = method;
1001                                 bestMatchCost = myCost;
1002                             }
1003                         }
1004                     }
1005                 }
1006             }
1007         }
1008         if (bestMatch != null) { //find a match
1009             cacheMethod(md, bestMatch);
1010         }
1011         return bestMatch;
1012     }

1047      * Returns the sum of the object transformation cost for each class in the source
1048      * argument list.
1049      *
1050      * @param srcArgs  The source arguments
1051      * @param destArgs The destination arguments
1052      *
1053      * @return The total transformation cost
1054      */
1055     private static float getTotalTransformationCost(Class[] srcArgs, Class[] destArgs) {
1056 
1057         float totalCost = 0.0f;
1058         for (int i = 0; i < srcArgs.length; i++) {
1059             Class srcClass, destClass;
1060             srcClass = srcArgs[i];
1061             destClass = destArgs[i];
1062             totalCost += getObjectTransformationCost(srcClass, destClass);
1063         }
1064 
1065         return totalCost;
1066     }

1068     /**
1069      * Gets the number of steps required needed to turn the source class into the
1070      * destination class. This represents the number of steps in the object hierarchy
1071      * graph.
1072      *
1073      * @param srcClass  The source class
1074      * @param destClass The destination class
1075      *
1076      * @return The cost of transforming an object
1077      */
1078     private static float getObjectTransformationCost(Class srcClass, Class destClass) {
1079         float cost = 0.0f;
1080         while (destClass != null && !destClass.equals(srcClass)) {
1081             if (destClass.isInterface() && isAssignmentCompatible(destClass, srcClass)) {
1082                 // slight penalty for interface match. 
1083                 // we still want an exact match to override an interface match, but  
1084                 // an interface match should override anything where we have to get a 
1085                 // superclass.
1086                 cost += 0.25f;
1087                 break;
1088             }
1089             cost++;
1090             destClass = destClass.getSuperclass();
1091         }
1092 
1093         /*
1094          * If the destination class is null, we've travelled all the way up to 
1095          * an Object match. We'll penalize this by adding 1.5 to the cost.
1096          */
1097         if (destClass == null) {
1098             cost += 1.5f;
1099         }
1100 
1101         return cost;
1102     }
