55   * @author Andres Almiray
56   * @since 2.0.0
57   */
58  @Named("lanterna")
59  @DependsOn({"core"})
60  @SuppressWarnings("rawtypes")
61  public class LanternaBuilderCustomizer extends AbstractBuilderCustomizer {
62      @SuppressWarnings("unchecked")
63      public LanternaBuilderCustomizer() {
64          Map<String, Factory> factories = new LinkedHashMap<>();
65          factories.put("action", new ActionFactory());
66          factories.put("actionListBox", new ActionListBoxFactory());
67          factories.put("actions", new CollectionFactory());
68          factories.put("application", new ApplicationFactory());
69          factories.put("bean", new ComponentFactory(Object.class, true));
70          factories.put("borderLayout", new LayoutFactory(BorderLayout.class));
71          factories.put("button", new ButtonFactory());
72          factories.put("checkBox", new CheckBoxFactory());
73          factories.put("container", new ComponentFactory(Component.class, false));
74          factories.put("emptySpace", new EmptySpaceFactory());
75          factories.put("hbox", new BoxFactory(Panel.Orientation.HORISONTAL));
76          factories.put("horisontalLayout", new LayoutFactory(HorisontalLayout.class));
77          factories.put("horizontalLayout", new LayoutFactory(HorisontalLayout.class));
78          factories.put("label", new LabelFactory());
79          factories.put("list", new CollectionFactory());
80          factories.put("panel", new PanelFactory());
81          factories.put("passwordBox", new PasswordBoxFactory());
82          factories.put("progressBar", new ProgressBarFactory());
83          factories.put("table", new TableFactory());
84          factories.put("textArea", new TextAreaFactory());
85          factories.put("textBox", new TextBoxFactory());
86          factories.put("vbox", new BoxFactory(Panel.Orientation.VERTICAL));
87          factories.put("verticalLayout", new LayoutFactory(VerticalLayout.class));
88          factories.put("widget", new ComponentFactory(Component.class, true));
89          setFactories(factories);
90  
91          Closure c = new Closure(this) {
92              @Override
93              public Object call(Object... args) {
94                  return handleIdAttribute(args);
95              }
96  
97              private Object handleIdAttribute(Object[] args) {
98                  FactoryBuilderSupport builder = (FactoryBuilderSupport) args[0];
99                  Object node = args[1];
100                 Map attributes = (Map) args[2];
101                 if (attributes.containsKey("id")) {
102                     String id = attributes.remove("id").toString();
103                     builder.setVariable(id, node);
104                 }
105                 return null;
106             }
107         };
108         setAttributeDelegates(newList(c));
109     }
110 }
