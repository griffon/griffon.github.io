18  package griffon.builder.swing;
19  
20  import griffon.builder.swing.factory.ApplicationFactory;
21  import griffon.inject.DependsOn;
22  import groovy.swing.SwingBuilder;
23  import groovy.util.Factory;
24  import org.codehaus.griffon.runtime.groovy.view.AbstractBuilderCustomizer;
25  
26  import javax.inject.Named;
27  import java.util.LinkedHashMap;
28  import java.util.Map;
29  
30  /**
31   * @author Andres Almiray
32   */
33  @Named("swing")
34  @DependsOn({"core"})
35  @SuppressWarnings("rawtypes")
36  public class SwingBuilderCustomizer extends AbstractBuilderCustomizer {
37      @SuppressWarnings("unchecked")
38      public SwingBuilderCustomizer() {
39          SwingBuilder builder = new SwingBuilder();
40  
41          Map<String, Factory> factories = new LinkedHashMap<>(builder.getFactories());
42          factories.put("application", new ApplicationFactory());
43          setFactories(factories);
44          setVariables(builder.getVariables());
45          setMethods(builder.getExplicitMethods());
46          setProps(builder.getExplicitProperties());
47          setAttributeDelegates(builder.getAttributeDelegates());
48          setPreInstantiateDelegates(builder.getPreInstantiateDelegates());
49          setPostInstantiateDelegates(builder.getPostInstantiateDelegates());
50          setPostNodeCompletionDelegates(builder.getPostNodeCompletionDelegates());
51          setDisposalClosures(builder.getDisposalClosures());
52          setMethodMissingDelegate(builder.getMethodMissingDelegate());
53          setPropertyMissingDelegate(builder.getPropertyMissingDelegate());
54      }
55  }
