218         requireNonNull(params, ERROR_PARAMS_NULL);

221         MethodDescriptorml#MethodDescriptor">MethodDescriptor target = new MethodDescriptor(eventHandler, argTypes);

241     @Override
242     @SuppressWarnings("unchecked")
243     public void addEventListener(@Nonnull Object listener) {
244         requireNonNull(listener, ERROR_LISTENER_NULL);
245         if (listener instanceof RunnableWithArgs) {
246             throw new IllegalArgumentException("Cannot add an event listener of type " + RunnableWithArgs.class.getName() +
247                 " because the target event name is missing. " + listener);
248         }
249         if (listener instanceof CallableWithArgs) {
250             throw new IllegalArgumentException("Cannot add an event listener of type " + CallableWithArgs.class.getName() +
251                 " because the target event name is missing. " + listener);
252         }
253 
254         if (listener instanceof Map) {
255             addEventListener((Map) listener);
256             return;
257         }
258 
259         if (!methodCache.isEventListener(listener.getClass())) {
260             return;
261         }
262 
263         boolean added = false;
264         for (String eventName : methodCache.fetchMethodMetadata(listener.getClass()).keySet()) {
265             eventName = eventName.substring(2); // cut off "on" from the name
266             List<Object> instances = instanceListeners.computeIfAbsent(eventName, (k) -> new ArrayList<>());
267             if (!instances.contains(listener)) {
268                 added = true;
269                 instances.add(listener);
270             }
271         }
272 
273         if (added) {
274             try {
275                 LOG.debug("Adding listener {}", listener);
276             } catch (UnsupportedOperationException uoe) {
277                 LOG.debug("Adding listener {}", listener.getClass().getName());
278             }
279         }
280     }

284         requireNonNull(listener, ERROR_LISTENER_NULL);
285         for (Map.Entry<String, Object> entry : listener.entrySet()) {
286             Object eventHandler = entry.getValue();
287             if (eventHandler instanceof RunnableWithArgs) {
288                 addEventListener(entry.getKey(), (RunnableWithArgs) eventHandler);
289             } else if (eventHandler instanceof CallableWithArgs) {
290                 addEventListener(entry.getKey(), (CallableWithArgs) eventHandler);
291             } else {
292                 throw new IllegalArgumentException("Unsupported functional event listener " + eventHandler);
293             }
294         }

297     @Override
298     @SuppressWarnings("unchecked")
299     public void removeEventListener(@Nonnull Object listener) {
300         requireNonNull(listener, ERROR_LISTENER_NULL);
301         if (listener instanceof RunnableWithArgs) {
302             throw new IllegalArgumentException("Cannot remove an event listener of type " + RunnableWithArgs.class.getName() +
303                 " because the target event name is missing. " + listener);
304         }
305         if (listener instanceof CallableWithArgs) {
306             throw new IllegalArgumentException("Cannot remove an event listener of type " + CallableWithArgs.class.getName() +
307                 " because the target event name is missing. " + listener);
308         }
309 
310         if (listener instanceof Map) {
311             removeEventListener((Map) listener);
312             return;
313         }
314 
315         boolean removed = false;
316         for (String eventName : methodCache.fetchMethodMetadata(listener.getClass()).keySet()) {
317             eventName = eventName.substring(2); // cut off "on" from the name
318             List<Object> instances = instanceListeners.get(eventName);
319             if (instances != null && instances.contains(listener)) {
320                 instances.remove(listener);
321                 removed = true;
322                 if (instances.isEmpty()) {
323                     instanceListeners.remove(eventName);
324                 }
325             }
326         }
327 
328         boolean nestedRemoved = removeNestedListeners(listener);
329 
330         if (removed || nestedRemoved) {
331             try {
332                 LOG.debug("Removing listener {}", listener);
333             } catch (UnsupportedOperationException uoe) {
334                 LOG.debug("Removing listener {}", listener.getClass().getName());
335             }
336         }
337     }

404     @Nonnull
405     @Override
406     public Collection<Object> getEventListeners() {
407         List<Object> listeners = new ArrayList<>();
408         synchronized (instanceListeners) {
409             Set<Object> instances = new HashSet<>();
410             for (List<Object> objects : instanceListeners.values()) {
411                 instances.addAll(objects);
412             }
413             listeners.addAll(instances);
414         }
415 
416         synchronized (functionalListeners) {
417             for (List<Object> objects : functionalListeners.values()) {
418                 listeners.addAll(objects);
419             }
420         }
421 
422         return unmodifiableCollection(listeners);
423     }

425     @Nonnull
426     @Override
427     public Collection<Object> getEventListeners(@Nonnull String eventName) {
428         requireNonBlank(eventName, ERROR_EVENT_NAME_BLANK);
429         List<Object> listeners = new ArrayList<>();
430         List<Object> instances = instanceListeners.get(eventName);
431         if (instances != null) { listeners.addAll(instances); }
432         instances = functionalListeners.get(eventName);
433         if (instances != null) { listeners.addAll(instances); }
434         return unmodifiableCollection(listeners);
435     }

437     protected Runnable buildPublisher(@Nonnull final String event, @Nonnull final List<?> params, @Nonnull final String mode) {
438         requireNonNull(event, ERROR_EVENT_NULL);
439         requireNonNull(params, ERROR_PARAMS_NULL);
440         requireNonBlank(mode, ERROR_MODE_BLANK);
441         return () -> {
442             String eventName = capitalize(event);
443             LOG.debug("Triggering event '{}' {}", eventName, mode);
444             String eventHandler = "on" + eventName;
445             // defensive copying to avoid CME during event dispatching
446             List<Object> listenersCopy = new ArrayList<>();
447             List<Object> instances = instanceListeners.get(eventName);
448             if (instances != null) {
449                 listenersCopy.addAll(instances);
450             }
451             synchronized (functionalListeners) {
452                 List<?> list = functionalListeners.get(eventName);
453                 if (list != null) {
454                     listenersCopy.addAll(list);
455                 }
456             }
457 
458             for (Object listener : listenersCopy) {
459                 if (listener instanceof RunnableWithArgs) {
460                     fireEvent((RunnableWithArgs) listener, params);
461                 } else if (listener instanceof CallableWithArgs) {
462                     fireEvent((CallableWithArgs<?>) listener, params);
463                 } else {
464                     fireEvent(listener, eventHandler, params);
465                 }
466             }
467         };
468     }

470     protected boolean removeNestedListeners(@Nonnull Object owner) {
471         requireNonNull(owner, ERROR_OWNER_NULL);
472 
473         boolean removed = false;
474         synchronized (functionalListeners) {
475             for (Map.Entry<String, List<Object>> event : functionalListeners.entrySet()) {
476                 String eventName = event.getKey();
477                 List<Object> listenerList = event.getValue();
478                 List<Object> toRemove = new ArrayList<>();
479                 for (Object listener : listenerList) {
480                     if (isNestedListener(listener, owner)) {
481                         toRemove.add(listener);
482                     }
483                 }
484                 removed = !toRemove.isEmpty();
485                 for (Object listener : toRemove) {
486                     LOG.debug("Removing listener {} on {}", listener.getClass().getName(), capitalize(eventName));
487                     listenerList.remove(listener);
488                 }
489             }
490         }
491 
492         return removed;
493     }

495     protected boolean isNestedListener(@Nonnull Object listener, @Nonnull Object owner) {
496         requireNonNull(listener, ERROR_LISTENER_NULL);
497         requireNonNull(owner, ERROR_OWNER_NULL);
498         Class<?> listenerClass = listener.getClass();
499         return (listenerClass.isMemberClass() || listenerClass.isAnonymousClass() || listenerClass.isLocalClass()) &&
500             owner.getClass().equals(listenerClass.getEnclosingClass()) &&
501             owner.equals(GriffonClassUtils.getFieldValue(listener, "this$0"));
502     }

508     protected static class MethodCache {
509         private final Map<Class<?>, Map<String, List<MethodInfo>>> methodMap = new ConcurrentHashMap<>();
510 
511         public boolean isEventListener(@Nonnull Class<?> klass) {
512             Map<String, List<MethodInfo>> methodMetadata = methodMap.get(klass);
513             if (methodMetadata == null) {
514                 methodMetadata = fetchMethodMetadata(klass);
515                 if (!methodMetadata.isEmpty()) {
516                     methodMap.put(klass, methodMetadata);
517                 } else {
518                     methodMetadata = null;
519                 }
520             }
521             return methodMetadata != null;
522         }
523 
524         @Nullable
525         public Method findMatchingMethodFor(@Nonnull Class<?> klass, @Nonnull MethodDescriptor target) {
526             Map<String, List<MethodInfo>> methodMetadata = methodMap.get(klass);
527 
528             List<MethodInfo> descriptors = methodMetadata.get(target.getName());
529             if (descriptors != null) {
530                 for (MethodInfo info : descriptors) {
531                     if (info.descriptor.matches(target)) {
532                         return info.method;
533                     }
534                 }
535             }
536 
537             return null;
538         }
539 
540         private Map<String, List<MethodInfo>> fetchMethodMetadata(Class<?> klass) {
541             Map<String, List<MethodInfo>> methodMetadata = new LinkedHashMap<>();
542 
543             for (Method method : klass.getMethods()) {
544                 MethodDescriptor descriptor = MethodDescriptor.forMethod(method);
545                 if (GriffonClassUtils.isEventHandler(descriptor)) {
546                     String methodName = method.getName();
547                     List<MethodInfo> descriptors = methodMetadata.computeIfAbsent(methodName, k -> new ArrayList<>());
548                     descriptors.add(new MethodInfo(descriptor, method));
549                 }
550             }
551 
552             return methodMetadata;
553         }
554     }

574     private static class DefaultThreadFactory implements ThreadFactory {
575         private final ThreadGroup group;
576         private final AtomicInteger threadNumber = new AtomicInteger(1);
577         private final String namePrefix;
578 
579         private DefaultThreadFactory(int eventRouterId) {
580             SecurityManager s = System.getSecurityManager();
581             group = (s != null) ? s.getThreadGroup() :
582                 Thread.currentThread().getThreadGroup();
583             namePrefix = "event-router-" + eventRouterId + "-thread-";
584         }
585 
586         public Thread newThread(Runnable r) {
587             Thread t = new Thread(group, r, namePrefix + threadNumber.getAndIncrement(), 0);
588             if (t.isDaemon()) { t.setDaemon(false); }
589             if (t.getPriority() != Thread.NORM_PRIORITY) { t.setPriority(Thread.NORM_PRIORITY); }
590             return t;
591         }
