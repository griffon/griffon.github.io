114     protected MVCGroup createMVCGroup(@Nonnull MVCGroupConfiguration configuration, @Nullable String mvcId, @Nonnull Map<String, Object> args) {
115         requireNonNull(configuration, ERROR_CONFIGURATION_NULL);
116         requireNonNull(args, ERROR_ARGS_NULL);
117 
118         mvcId = resolveMvcId(configuration, mvcId);
119         checkIdIsUnique(mvcId, configuration);
120 
121         LOG.debug("Building MVC group '{}' with name '{}'", configuration.getMvcType(), mvcId);
122         Map<String, Object> argsCopy = copyAndConfigureArguments(args, configuration, mvcId);
123 
124         // figure out what the classes are
125         Map<String, ClassHolder> classMap = new LinkedHashMap<>();
126         for (Map.Entry<String, String> memberEntry : configuration.getMembers().entrySet()) {
127             String memberType = memberEntry.getKey();
128             String memberClassName = memberEntry.getValue();
129             selectClassesPerMember(memberType, memberClassName, classMap);
130         }
131 
132         boolean isEventPublishingEnabled = getApplication().getEventRouter().isEventPublishingEnabled();
133         getApplication().getEventRouter().setEventPublishingEnabled(isConfigFlagEnabled(configuration, CONFIG_KEY_EVENTS_INSTANTIATION));
134         Map<String, Object> instances = new LinkedHashMap<>();
135         List<Object> injectedInstances = new ArrayList<>();
136 
137         try {
138             InjectionUnitOfWork.start();
139         } catch (IllegalStateException ise) {
140             throw new MVCGroupInstantiationException("Can not instantiate MVC group '" + configuration.getMvcType() + "' with id '" + mvcId + "'", configuration.getMvcType(), mvcId, ise);
141         }
142 
143         try {
144             instances.putAll(instantiateMembers(classMap, argsCopy));
145         } finally {
146             getApplication().getEventRouter().setEventPublishingEnabled(isEventPublishingEnabled);
147             try {
148                 injectedInstances.addAll(InjectionUnitOfWork.finish());
149             } catch (IllegalStateException ise) {
150                 throw new MVCGroupInstantiationException("Can not instantiate MVC group '" + configuration.getMvcType() + "' with id '" + mvcId + "'", configuration.getMvcType(), mvcId, ise);
151             }
152         }
153 
154         MVCGroupml#MVCGroup">MVCGroup group = newMVCGroup(configuration, mvcId, instances, (MVCGroup) args.get(KEY_PARENT_GROUP));
155         adjustMvcArguments(group, argsCopy);
156 
157         boolean fireEvents = isConfigFlagEnabled(configuration, CONFIG_KEY_EVENTS_LIFECYCLE);
158         if (fireEvents) {
159             getApplication().getEventRouter().publishEvent(ApplicationEvent.INITIALIZE_MVC_GROUP.getName(), asList(configuration, group));
160         }
161 
162         // special case -- controllers are added as application listeners
163         if (isConfigFlagEnabled(group.getConfiguration(), CONFIG_KEY_EVENTS_LISTENER)) {
164             GriffonController controller = group.getController();
165             if (controller != null) {
166                 getApplication().getEventRouter().addEventListener(controller);
167             }
168         }
169 
170         // mutually set each other to the available fields and inject args
171         fillReferencedProperties(group, argsCopy);
172 
173         doAddGroup(group);
174 
175         initializeMembers(group, argsCopy);
176         if (group instanceof AbstractMVCGroup) {
177             ((AbstractMVCGroup) group).getInjectedInstances().addAll(injectedInstances);
178         }
179 
180         if (fireEvents) {
181             getApplication().getEventRouter().publishEvent(ApplicationEvent.CREATE_MVC_GROUP.getName(), singletonList(group));
182         }
183 
184         return group;
185     }

252     @Nonnull
253     protected Map<String, Object> instantiateMembers(@Nonnull Map<String, ClassHolder> classMap, @Nonnull Map<String, Object> args) {
254         // instantiate the parts
255         Map<String, Object> instanceMap = new LinkedHashMap<>();
256         for (Map.Entry<String, ClassHolder> classEntry : classMap.entrySet()) {
257             String memberType = classEntry.getKey();
258             if (args.containsKey(memberType)) {
259                 // use provided value, even if null
260                 instanceMap.put(memberType, args.get(memberType));
261             } else {
262                 // otherwise create a new value
263                 ClassHolder classHolder = classEntry.getValue();
264                 if (classHolder.artifactClass != null) {
265                     Class<? extends GriffonArtifact> memberClass = classHolder.artifactClass;
266                     ArtifactManager artifactManager = getApplication().getArtifactManager();
267                     GriffonClass griffonClass = artifactManager.findGriffonClass(memberClass);
268                     GriffonArtifact instance = artifactManager.newInstance(griffonClass);
269                     instanceMap.put(memberType, instance);
270                     args.put(memberType, instance);
271                 } else {
272                     Class<?> memberClass = classHolder.regularClass;
273                     try {
274                         Object instance = instantiator.instantiate(memberClass);
275                         instanceMap.put(memberType, instance);
276                         args.put(memberType, instance);
277                     } catch (RuntimeException e) {
278                         LOG.error("Can't create member {} with {}", memberType, memberClass);
279                         throw new NewInstanceException(memberClass, e);
280                     }
281                 }
282             }
283         }
284         return instanceMap;
285     }

320     protected abstract static class InjectionPoint {
321         protected final String name;
322         protected final boolean nullable;
323         protected final Kind kind;
324         protected final Class<?> type;
325         protected final String format;
326         protected final Class<? extends PropertyEditor> editor;
327 
328         protected InjectionPoint(String name, boolean nullable, Kind kind, Class<?> type, String format, Class<? extends PropertyEditor> editor) {
329             this.name = name;
330             this.nullable = nullable;
331             this.kind = kind;
332             this.type = type; this.format = format;
333             this.editor = editor;
334         }
335 
336         protected enum Kind {
337             MEMBER,
338             CONTEXTUAL,
339             OTHER
340         }
341 
342         protected abstract void apply(@Nonnull MVCGroup group, @Nonnull String memberType, @Nonnull Object instance, @Nonnull Map<String, Object> args);
343 
344         @Nonnull
345         protected Object convertValue(@Nonnull Class<?> type, @Nonnull Object value, @Nullable String format, @Nonnull Class<? extends PropertyEditor> editor) {
346             requireNonNull(type, ERROR_TYPE_NULL);
347             requireNonNull(value, ERROR_VALUE_NULL);
348 
349             PropertyEditor propertyEditor = resolvePropertyEditor(type, format, editor);
350             if (isNoopPropertyEditor(propertyEditor.getClass())) { return value; }
351             if (value instanceof CharSequence) {
352                 propertyEditor.setAsText(String.valueOf(value));
353             } else {
354                 propertyEditor.setValue(value);
355             }
356             return propertyEditor.getValue();
357         }
358 
359         @Nonnull
360         protected PropertyEditor resolvePropertyEditor(@Nonnull Class<?> type, @Nullable String format, @Nonnull Class<? extends PropertyEditor> editor) {
361             requireNonNull(type, ERROR_TYPE_NULL);
362 
363             PropertyEditor propertyEditor = null;
364             if (isNoopPropertyEditor(editor)) {
365                 propertyEditor = findEditor(type);
366             } else {
367                 try {
368                     propertyEditor = editor.newInstance();
369                 } catch (InstantiationException | IllegalAccessException e) {
370                     throw new GriffonException("Could not instantiate editor with " + editor, e);
371                 }
372             }
373 
374             if (propertyEditor instanceof ExtendedPropertyEditor) {
375                 ((ExtendedPropertyEditor) propertyEditor).setFormat(format);
376             }
377             return propertyEditor;
378         }
379 
380         protected boolean isNoopPropertyEditor(@Nonnull Class<? extends PropertyEditor> editor) {
381             return PropertyEditorResolver.NoopPropertyEditor.class.isAssignableFrom(editor);
382         }
383     }

385     protected static class FieldInjectionPoint extends InjectionPoint {
386         protected final Field field;
387 
388         protected FieldInjectionPoint(String name, boolean nullable, Kind kind, Class<?> type, Field field, String format, Class<? extends PropertyEditor> editor) {
389             super(name, nullable, kind, type, format, editor);
390             this.field = field;
391         }
392 
393         @Override
394         protected void apply(@Nonnull MVCGroup group, @Nonnull String memberType, @Nonnull Object instance, @Nonnull Map<String, Object> args) {
395             String[] keys = namesFor(field);
396             Object argValue = args.get(name);
397 
398             if (kind == Kind.CONTEXTUAL) {
399                 for (String key : keys) {
400                     if (group.getContext().containsKey(key)) {
401                         argValue = group.getContext().get(key);
402                         break;
403                     }
404                 }
405             }
406 
407             try {
408                 if (argValue == null) {
409                     if (!nullable) {
410                         if (kind == Kind.CONTEXTUAL) {
411                             throw new IllegalStateException("Could not find an instance of type " +
412                                 field.getType().getName() + " under keys '" + Arrays.toString(keys) +
413                                 "' in the context of MVCGroup[" + group.getMvcType() + ":" + group.getMvcId() +
414                                 "] to be injected on field '" + field.getName() +
415                                 "' in " + kind + " (" + resolveMemberClass(instance).getName() + "). Field does not accept null values.");
416                         } else if (kind == Kind.MEMBER) {
417                             throw new IllegalStateException("Could not inject argument on field '"
418                                 + name + "' in " + memberType + " (" + resolveMemberClass(instance).getName() +
419                                 "). Field does not accept null values.");
420                         }
421                     }
422                     return;
423                 } else if (kind == Kind.MEMBER && (!isNoopPropertyEditor(editor) || !type.isAssignableFrom(argValue.getClass()))) {
424                     argValue = convertValue(type, argValue, format, editor);
425                 }
426 
427                 setFieldValue(instance, name, argValue);
428                 if (kind == Kind.OTHER) {
429                     LOG.warn("Field '" + name + "' in " + memberType + " (" + resolveMemberClass(instance).getName() +
430                         ") must be annotated with @" + MVCMember.class.getName() + ".");
431                 }
432             } catch (Exception e) {
433                 throw new MVCGroupInstantiationException("Unexpected error when setting value for " + resolveMemberClass(instance).getName() + "." + field.getName(), group.getMvcType(), group.getMvcId(), e);
434             }
435         }
436     }

500     protected void fillReferencedProperties(@Nonnull MVCGroup group, @Nonnull Map<String, Object> args) {
501         for (Map.Entry<String, Object> memberEntry : group.getMembers().entrySet()) {
502             String memberType = memberEntry.getKey();

504 
505             Map<String, Object> argsCopy = new LinkedHashMap<>(args);
506 
507             Map<String, Field> fields = new LinkedHashMap<>();
508             for (Field field : getAllDeclaredFields(resolveMemberClass(member))) {
509                 fields.put(field.getName(), field);
510             }
511             Map<String, InjectionPoint> injectionPoints = new LinkedHashMap<>();
512             for (PropertyDescriptor descriptor : getPropertyDescriptors(resolveMemberClass(member))) {
513                 Method method = descriptor.getWriteMethod();
514                 if (method == null || isInjectable(method)) { continue; }
515                 Class<?> type = parameterTypeAt(method, 0);
516                 boolean nullable = method.getAnnotation(Nonnull.class) == null && findAnnotation(annotationsOfMethodParameter(method, 0), Nonnull.class) == null;
517                 InjectionPoint.Kind kind = resolveKind(method);
518                 String format = resolveFormat(method);
519                 Class<? extends PropertyEditor> editor = resolveEditor(method);
520                 Field field = fields.get(descriptor.getName());
521                 if (field != null && kind == InjectionPoint.Kind.OTHER) {
522                     kind = resolveKind(field);
523                     nullable = field.getAnnotation(Nonnull.class) == null;
524                     type = field.getType();
525                     format = resolveFormat(field);
526                     editor = resolveEditor(field);
527                 }
528                 injectionPoints.put(descriptor.getName(), new MethodInjectionPoint(descriptor.getName(), nullable, kind, type, method, format, editor));
529             }
530 
531             for (Field field : getAllDeclaredFields(resolveMemberClass(member))) {
532                 if (Modifier.isStatic(field.getModifiers()) || isInjectable(field)) { continue; }
533                 if (!injectionPoints.containsKey(field.getName())) {
534                     boolean nullable = field.getAnnotation(Nonnull.class) == null;
535                     InjectionPoint.Kind kind = resolveKind(field);
536                     Class<?> type = field.getType();
537                     String format = resolveFormat(field);
538                     Class<? extends PropertyEditor> editor = resolveEditor(field);
539                     injectionPoints.put(field.getName(), new FieldInjectionPoint(field.getName(), nullable, kind, type, field, format, editor));
540                 }
541             }
542 
543             for (InjectionPoint ip : injectionPoints.values()) {
544                 ip.apply(group, memberType, member, args);
545                 argsCopy.remove(ip.name);
546             }

553                         ") must be annotated with @" + MVCMember.class.getName() + ".");
554                 } catch (PropertyException ignored) {

605     public void destroyMVCGroup(@Nonnull String mvcId) {
606         MVCGroup group = findGroup(mvcId);
607         LOG.debug("Group '{}' points to {}", mvcId, group);
608 
609         if (group == null) { return; }
610 
611         LOG.debug("Destroying MVC group identified by '{}'", mvcId);
612 
613         if (isConfigFlagEnabled(group.getConfiguration(), CONFIG_KEY_EVENTS_LISTENER)) {
614             GriffonController controller = group.getController();
615             if (controller != null) {
616                 getApplication().getEventRouter().removeEventListener(controller);
617             }
618         }
619 
620         boolean fireDestructionEvents = isConfigFlagEnabled(group.getConfiguration(), CONFIG_KEY_EVENTS_DESTRUCTION);
621 
622         destroyMembers(group, fireDestructionEvents);
623 
624         doRemoveGroup(group);
625         group.destroy();
626 
627         if (isConfigFlagEnabled(group.getConfiguration(), CONFIG_KEY_EVENTS_LIFECYCLE)) {
628             getApplication().getEventRouter().publishEvent(ApplicationEvent.DESTROY_MVC_GROUP.getName(), singletonList(group));
629         }
630     }

632     protected void destroyMembers(@Nonnull MVCGroup group, boolean fireDestructionEvents) {
633         for (Map.Entry<String, Object> memberEntry : group.getMembers().entrySet()) {
634             Object member = memberEntry.getValue();
635             if (member instanceof GriffonArtifact) {
636                 destroyArtifactMember(memberEntry.getKey(), (GriffonArtifact) member, fireDestructionEvents);
637             } else {
638                 destroyNonArtifactMember(memberEntry.getKey(), member, fireDestructionEvents);
639             }
640 
641         }
642 
643         if (group instanceof AbstractMVCGroup) {
644             List<Object> injectedInstances = ((AbstractMVCGroup) group).getInjectedInstances();
645             for (Object instance : injectedInstances) {
646                 getApplication().getInjector().release(instance);
647             }
648             injectedInstances.clear();
649         }
650     }

680     protected void destroyContextualMemberProperties(@Nonnull String type, @Nonnull GriffonArtifact member) {
681         for (Field field : getAllDeclaredFields(member.getTypeClass())) {
682             if (isContextual(field) && !field.getType().isPrimitive()) {
683                 try {
684                     setFieldValue(member, field.getName(), null);
685                 } catch (FieldException e) {
686                     throw new IllegalStateException("Could not nullify field '" +
687                         field.getName() + "' in " + type + " (" + member.getTypeClass().getName() + ")", e);
688                 }
689             }
690         }
691     }
