59  /**
60   * Utility class for bootstrapping an application.
61   *
62   * @author Danno Ferrin
63   * @author Andres Almiray
64   */
65  public class DefaultApplicationConfigurer implements ApplicationConfigurer {
66      private static final Logger LOG = LoggerFactory.getLogger(DefaultApplicationConfigurer.class);
67  
68      private static final String ERROR_APPLICATION_NULL = "Argument 'application' must not be null";
69      private static final String KEY_APP_LIFECYCLE_HANDLER_DISABLE = "application.lifecycle.handler.disable";
70      private static final String KEY_GRIFFON_CONTROLLER_ACTION_HANDLER_ORDER = "griffon.controller.action.handler.order";
71  
72      private final Object lock = new Object();
73      private final GriffonApplication application;
74      @GuardedBy("lock")
75      private boolean initialized;
76  
77      @Inject
78      public DefaultApplicationConfigurer(@Nonnull GriffonApplication application) {
79          this.application = requireNonNull(application, ERROR_APPLICATION_NULL);
80      }
81  
82      @Override
83      public final void init() {
84          synchronized (lock) {
85              if (!initialized) {
86                  doInitialize();
87                  initialized = true;
88              }
89          }
90      }
91  
92      @Override
93      public void runLifecycleHandler(@Nonnull Lifecycle lifecycle) {
94          requireNonNull(lifecycle, "Argument 'lifecycle' must not be null");
95  
96          boolean skipHandler = application.getConfiguration().getAsBoolean(KEY_APP_LIFECYCLE_HANDLER_DISABLE, false);
97          if (skipHandler) {
98              LOG.info("Lifecycle handler '{}' has been disabled. SKIPPING.", lifecycle.getName());
99              return;
100         }
101 
102         LifecycleHandler handler;
103         try {
104             handler = application.getInjector().getInstance(LifecycleHandler.class, named(lifecycle.getName()));
105         } catch (Exception e) {
106             // the script must not exist, do nothing
107             //LOGME - may be because of chained failures
108             return;
109         }
110 
111         handler.execute();
112     }
113 
114     protected void doInitialize() {
115         initializeEventHandler();
116 
117         event(ApplicationEvent.BOOTSTRAP_START, singletonList(application));
118 
119         initializePropertyEditors();
120         initializeResourcesInjector();
121         initializeConfigurationManager();
122         runLifecycleHandler(Lifecycle.INITIALIZE);
123         applyPlatformTweaks();
124         initializeAddonManager();
125         initializeMvcManager();
126         initializeActionManager();
127         initializeArtifactManager();
128 
129         event(ApplicationEvent.BOOTSTRAP_END, singletonList(application));
130     }
131 
132     protected void initializeEventHandler() {
133         Collection<EventHandler> handlerInstances = application.getInjector().getInstances(EventHandler.class);
134         Map<String, EventHandler> sortedHandlers = sortByDependencies(handlerInstances, "EventHandler", "handler");
135         for (EventHandler handler : sortedHandlers.values()) {
136             application.getEventRouter().addEventListener(handler);
137         }
138     }
139 
140     protected void event(@Nonnull ApplicationEvent event, @Nullable List<?> args) {
141         application.getEventRouter().publishEvent(event.getName(), args);
142     }
143 
144     protected void initializePropertyEditors() {
145         ServiceLoaderUtils.load(applicationClassLoader().get(), "META-INF/editors/", PropertyEditor.class, (classLoader, type, line) -> {
146             try {
147                 String[] parts = line.trim().split("=");
148                 Class<?> targetType = loadClass(parts[0].trim(), classLoader);
149                 Class<? extends PropertyEditor> editorClass = (Class<? extends PropertyEditor>) loadClass(parts[1].trim(), classLoader);
150 
151                 // Editor must have a no-args constructor
152                 // CCE means the class can not be used
153                 editorClass.newInstance();
154                 PropertyEditorResolver.registerEditor(targetType, editorClass);
155                 LOG.debug("Registering {} as editor for {}", editorClass.getName(), targetType.getName());
156             } catch (Exception e) {
157                 if (LOG.isWarnEnabled()) {
158                     LOG.warn("Could not load " + type.getName() + " with " + line, sanitize(e));
159                 }
160             }
161         });
162 
163         Class<?>[][] pairs = new Class<?>[][]{
164             new Class<?>[]{Boolean.class, Boolean.TYPE},
165             new Class<?>[]{Byte.class, Byte.TYPE},
166             new Class<?>[]{Short.class, Short.TYPE},
167             new Class<?>[]{Integer.class, Integer.TYPE},
168             new Class<?>[]{Long.class, Long.TYPE},
169             new Class<?>[]{Float.class, Float.TYPE},
170             new Class<?>[]{Double.class, Double.TYPE}
171         };
172 
173         for (Class<?>[] pair : pairs) {
174             PropertyEditor editor = PropertyEditorResolver.findEditor(pair[0]);
175             LOG.debug("Registering {} as editor for {}", editor.getClass().getName(), pair[1].getName());
176             PropertyEditorResolver.registerEditor(pair[1], editor.getClass());
177         }
178     }
179 
180     protected void initializeResourcesInjector() {
181         final ResourceInjector injector = application.getResourceInjector();
182         application.getEventRouter().addEventListener(ApplicationEvent.NEW_INSTANCE.getName(), args -> {
183             injector.injectResources(args[1]);
184         });
185     }
186 
187     protected void initializeArtifactManager() {
188         Injector<?> injector = application.getInjector();
189         ArtifactManager artifactManager = application.getArtifactManager();
190         for (ArtifactHandler<?> artifactHandler : injector.getInstances(ArtifactHandler.class)) {
191             artifactManager.registerArtifactHandler(artifactHandler);
192         }
193         artifactManager.loadArtifactMetadata();
194     }
195 
196     protected void applyPlatformTweaks() {
197         PlatformHandler platformHandler = application.getInjector().getInstance(PlatformHandler.class);
198         platformHandler.handle(application);
199     }
200 
201     protected void initializeConfigurationManager() {
202         application.getConfigurationManager();
203     }
204 
205     protected void initializeAddonManager() {
206         application.getAddonManager().initialize();
207     }
208 
209     @SuppressWarnings("unchecked")
210     protected void initializeMvcManager() {
211         Map<String, MVCGroupConfiguration> configurations = new LinkedHashMap<>();
212         Map<String, Map<String, Object>> mvcGroups = application.getConfiguration().get("mvcGroups", Collections.<String, Map<String, Object>>emptyMap());
213         if (mvcGroups != null) {
214             for (Map.Entry<String, Map<String, Object>> groupEntry : mvcGroups.entrySet()) {
215                 String type = groupEntry.getKey();
216                 LOG.debug("Adding MVC group {}", type);
217                 Map<String, Object> members = groupEntry.getValue();
218                 Map<String, Object> configMap = new LinkedHashMap<>();
219                 Map<String, String> membersCopy = new LinkedHashMap<>();
220                 for (Map.Entry<String, Object> entry : members.entrySet()) {
221                     String key = String.valueOf(entry.getKey());
222                     if ("config".equals(key) && entry.getValue() instanceof Map) {
223                         configMap = (Map<String, Object>) entry.getValue();
224                     } else {
225                         membersCopy.put(key, String.valueOf(entry.getValue()));
226                     }
227                 }
228                 configurations.put(type, application.getMvcGroupManager().newMVCGroupConfiguration(type, membersCopy, configMap));
229             }
230         }
231 
232         application.getMvcGroupManager().initialize(configurations);
233     }
234 
235     protected void initializeActionManager() {
236         if (application.getActionManager() instanceof NoopActionManager) {
237             return;
238         }
239 
240         application.getEventRouter().addEventListener(ApplicationEvent.NEW_INSTANCE.getName(), args -> {
241             Class<?> klass = (Class) args[0];
242             if (GriffonController.class.isAssignableFrom(klass)) {
243                 application.getActionManager().createActions((GriffonController) args[1]);
244             }
245         });
246 
247         Injector<?> injector = application.getInjector();
248         Collection<ActionHandler> handlerInstances = injector.getInstances(ActionHandler.class);
249         List<String> handlerOrder = application.getConfiguration().get(KEY_GRIFFON_CONTROLLER_ACTION_HANDLER_ORDER, Collections.<String>emptyList());
250         Map<String, ActionHandler> sortedHandlers = sortByDependencies(handlerInstances, ActionHandler.SUFFIX, "handler", handlerOrder);
251 
252         for (ActionHandler handler : sortedHandlers.values()) {
253             application.getActionManager().addActionHandler(handler);
254         }
255 
256         Collection<ActionInterceptor> interceptorInstances = injector.getInstances(ActionInterceptor.class);
257         if (!interceptorInstances.isEmpty()) {
258             application.getLog().error(ActionInterceptor.class.getName() + " has been deprecated and is no longer supported");
259             throw new UnsupportedOperationException(ActionInterceptor.class.getName() + " has been deprecated and is no longer supported");
260         }
261     }
262 
263     protected Class<?> loadClass(@Nonnull String className, @Nonnull ClassLoader classLoader) throws ClassNotFoundException {
264         ClassNotFoundException cnfe;
265 
266         ClassLoader cl = DefaultApplicationConfigurer.class.getClassLoader();
267         try {
268             return cl.loadClass(className);
269         } catch (ClassNotFoundException e) {
270             cnfe = e;
271         }
272 
273         cl = classLoader;
274         try {
275             return cl.loadClass(className);
276         } catch (ClassNotFoundException e) {
277             cnfe = e;
278         }
279 
280         throw cnfe;
281     }
282 
283     private ApplicationClassLoader applicationClassLoader() {
284         return application.getInjector().getInstance(ApplicationClassLoader.class);
285     }
