18  package org.codehaus.griffon.runtime.core;
19  
20  import griffon.core.ApplicationBootstrapper;
21  import griffon.core.GriffonApplication;
22  import griffon.core.artifact.GriffonService;
23  import griffon.core.env.GriffonEnvironment;
24  import griffon.core.injection.Binding;
25  import griffon.core.injection.Injector;
26  import griffon.core.injection.InjectorFactory;
27  import griffon.core.injection.Key;
28  import griffon.core.injection.Module;
29  import griffon.util.GriffonClassUtils;
30  import org.codehaus.griffon.runtime.core.injection.AbstractModule;
31  import org.slf4j.Logger;
32  import org.slf4j.LoggerFactory;
33  
34  import javax.annotation.Nonnull;
35  import java.util.ArrayList;
36  import java.util.Collection;
37  import java.util.Iterator;
38  import java.util.LinkedHashMap;
39  import java.util.List;
40  import java.util.Map;
41  import java.util.ServiceLoader;
42  
43  import static griffon.core.GriffonExceptionHandler.sanitize;
44  import static griffon.util.AnnotationUtils.sortByDependencies;
45  import static griffon.util.ServiceLoaderUtils.load;
46  import static java.util.Collections.unmodifiableCollection;
47  import static java.util.Objects.requireNonNull;
48  
49  /**
50   * @author Andres Almiray
51   * @since 2.0.0
52   */
53  public abstract class AbstractApplicationBootstrapper implements ApplicationBootstrapper {
54      private static final Logger LOG = LoggerFactory.getLogger(AbstractApplicationBootstrapper.class);
55      private static final String INJECTOR = "injector";
56      private static final String GRIFFON_PATH = "META-INF/griffon";
57      private static final String PROPERTIES = ".properties";
58      protected final GriffonApplication application;
59  
60      public AbstractApplicationBootstrapper(@Nonnull GriffonApplication application) {
61          this.application = requireNonNull(application, "Argument 'application' must not be null");
62      }
63  
64      @Override
65      public void bootstrap() throws Exception {
66          // 1 initialize environment settings
67          LOG.info("Griffon {}", GriffonEnvironment.getGriffonVersion());
68          LOG.info("Build: {}", GriffonEnvironment.getBuildDateTime());
69          LOG.info("Revision: {}", GriffonEnvironment.getBuildRevision());
70          LOG.info("JVM: {}", GriffonEnvironment.getJvmVersion());
71          LOG.info("OS: {}", GriffonEnvironment.getOsVersion());
72  
73          // 2 create bindings
74          LOG.debug("Creating module bindings");
75          Iterable<Binding<?>> bindings = createBindings();
76  
77          if (LOG.isTraceEnabled()) {
78              for (Binding<?> binding : bindings) {
79                  LOG.trace(binding.toString());
80              }
81          }
82  
83          // 3 create injector
84          LOG.debug("Creating application injector");
85          createInjector(bindings);
86      }
87  
88      @Override
89      public void run() {
90          application.initialize();
91          application.startup();
92          application.ready();
93      }
94  
95      @Nonnull
96      protected Iterable<Binding<?>> createBindings() {
97          Map<Key, Binding<?>> map = new LinkedHashMap<>();
98  
99          List<Module> modules = new ArrayList<>();
100         createApplicationModule(modules);
101         createArtifactsModule(modules);
102         collectModuleBindings(modules);
103 
104         for (Module module : modules) {
105             for (Binding<?> binding : module.getBindings()) {
106                 map.put(Key.of(binding), binding);
107             }
108         }
109 
110         return unmodifiableCollection(map.values());
111     }
112 
113     protected void createArtifactsModule(@Nonnull List<Module> modules) {
114         final List<Class<?>> classes = new ArrayList<>();
115         load(getClass().getClassLoader(), GRIFFON_PATH, path -> !path.endsWith(PROPERTIES), (classLoader, line) -> {
116             line = line.trim();
117             try {
118                 classes.add(classLoader.loadClass(line));
119             } catch (ClassNotFoundException e) {
120                 LOG.warn("'" + line + "' could not be resolved as a Class");
121             }
122         });
123 
124         modules.add(new AbstractModule() {
125             @Override
126             protected void doConfigure() {
127                 for (Class<?> clazz : classes) {
128                     if (GriffonService.class.isAssignableFrom(clazz)) {
129                         bind(clazz).asSingleton();
130                     } else {
131                         bind(clazz);
132                     }
133                 }
134             }
135         });
136     }
137 
138     protected void createApplicationModule(@Nonnull List<Module> modules) {
139         modules.add(new AbstractModule() {
140             @Override
141             protected void doConfigure() {
142                 bind(GriffonApplication.class)
143                     .toInstance(application);
144             }
145         });
146     }
147 
148     protected void collectModuleBindings(@Nonnull Collection<Module> modules) {
149         List<Module> moduleInstances = loadModules();
150         moduleInstances.add(0, new DefaultApplicationModule());
151         Map<String, Module> sortedModules = sortModules(moduleInstances);
152         for (Map.Entry<String, Module> entry : sortedModules.entrySet()) {
153             LOG.debug("Loading module bindings from {}:{}", entry.getKey(), entry.getValue());
154             modules.add(entry.getValue());
155         }
156     }
157 
158     @Nonnull
159     protected Map<String, Module> sortModules(@Nonnull List<Module> moduleInstances) {
160         return sortByDependencies(moduleInstances, "Module", "module");
161     }
162 
163     @Nonnull
164     protected abstract List<Module> loadModules();
165 
166     private void createInjector(@Nonnull Iterable<Binding<?>> bindings) throws Exception {
167         ServiceLoader<InjectorFactory> serviceLoader = ServiceLoader.load(InjectorFactory.class);
168         try {
169             Iterator<InjectorFactory> iterator = serviceLoader.iterator();
170             InjectorFactory injectorFactory = iterator.next();
171             LOG.debug("Injector will be created by {}", injectorFactory);
172             Injector<?> injector = injectorFactory.createInjector(application, bindings);
173             GriffonClassUtils.setProperty(application, INJECTOR, injector);
174         } catch (Exception e) {
175             LOG.error("An error occurred while initializing the injector", sanitize(e));
176             throw e;
177         }
178     }
179 }
