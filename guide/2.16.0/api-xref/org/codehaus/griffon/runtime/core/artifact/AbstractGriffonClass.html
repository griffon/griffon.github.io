41  /**
42   * Abstract base class for Griffon types that provides common functionality for
43   * evaluating conventions within classes
44   *
45   * @author Steven Devijver (Grails 0.1)
46   * @author Graeme Rocher (Grails 0.1)
47   * @author Andres Almiray
48   * @since 2.0.0
49   */
50  public abstract class AbstractGriffonClass implements GriffonClass {
51      private static final String ERROR_NAME_BLANK = "Argument 'name' must not be blank";
52      private static final String ERROR_ARTIFACT_TYPE_BLANK = "Argument 'artifactType' must not be blank";
53      private static final String ERROR_TYPE_NULL = "Argument 'type' must not be null";
54      private static final String ERROR_APPLICATION_NULL = "Argument 'application' must not be null";
55  
56      private final GriffonApplication application;
57      private final Class<?> clazz;
58      private final String artifactType;
59      private final String fullName;
60      private final String name;
61      private final String packageName;
62      private final String naturalName;
63      private final String shortName;
64      private final String propertyName;
65      private final String logicalPropertyName;
66      private final ClassPropertyFetcher classPropertyFetcher;
67  
68      protected final Set<String> eventsCache = new TreeSet<>();
69      protected final Logger log;
70  
71      public AbstractGriffonClass(@Nonnull GriffonApplication application, @Nonnull Class<?> type, @Nonnull String artifactType, @Nonnull String trailingName) {
72          this.application = requireNonNull(application, ERROR_APPLICATION_NULL);
73          this.clazz = requireNonNull(type, ERROR_TYPE_NULL);
74          this.artifactType = requireNonBlank(artifactType, ERROR_ARTIFACT_TYPE_BLANK).trim();
75          trailingName = isBlank(trailingName) ? "" : trailingName.trim();
76          fullName = type.getName();
77          log = LoggerFactory.getLogger(getClass().getSimpleName() + "[" + fullName + "]");
78          packageName = GriffonClassUtils.getPackageName(type);
79          naturalName = GriffonNameUtils.getNaturalName(type.getName());
80          shortName = GriffonClassUtils.getShortClassName(type);
81          name = GriffonNameUtils.getLogicalName(type, trailingName);
82          propertyName = getPropertyNameRepresentation(shortName);
83          if (isBlank(name)) {
84              logicalPropertyName = propertyName;
85          } else {
86              logicalPropertyName = getPropertyNameRepresentation(name);
87          }
88          classPropertyFetcher = forClass(type);
89      }
90

172         return "Artifact[" + artifactType + "] > " + getName();

175     public boolean equals(@Nullable Object obj) {
176         if (this == obj) return true;
177         if (obj == null) return false;
178         if (!obj.getClass().getName().equals(getClass().getName()))
179             return false;

183     }

185     public int hashCode() {
186         return clazz.hashCode() + artifactType.hashCode();
187     }

189     public void resetCaches() {
190         eventsCache.clear();
191     }
192 
193     @Nonnull
194     public String[] getEventNames() {
195         if (eventsCache.isEmpty()) {
196             for (Method method : getClazz().getMethods()) {
197                 String methodName = method.getName();
198                 if (!eventsCache.contains(methodName) &&
199                     isPlainMethod(method) &&
200                     isEventHandler(methodName)) {
201                     eventsCache.add(methodName.substring(2));
202                 }
203             }
204         }
205 
206         return eventsCache.toArray(new String[eventsCache.size()]);
207     }

209     /**
210      * Returns an array of property names that are backed by a filed with a matching
211      * name.<p>
212      * Fields must be private and non-static. Names will be returned in the order
213      * they are declared in the class, starting from the deepest class in the
214      * class hierarchy up to the topmost superclass != null
215      */
216     public String[] getPropertiesWithFields() {
217         return classPropertyFetcher.getPropertiesWithFields();
218     }
219 
220     public Class<?> getPropertyType(String name) {
221         return classPropertyFetcher.getPropertyType(name);
222     }
223 
224     public boolean isReadableProperty(String name) {
225         return classPropertyFetcher.isReadableProperty(name);
226     }
227 
228     /**
229      * <p>Looks for a property of the reference instance with a given name and type.</p>
230      * <p>If found its value is returned. We follow the Java bean conventions with augmentation for groovy support
231      * and static fields/properties. We will therefore match, in this order:
232      * </p>
233      * <ol>
234      * <li>Public static field
235      * <li>Public static property with getter method
236      * <li>Standard public bean property (with getter or just public field, using normal introspection)
237      * </ol>
238      *
239      * @return property value or null if no property or static field was found
240      */
241     protected Object getPropertyOrStaticPropertyOrFieldValue(@SuppressWarnings("hiding") @Nonnull String name, @Nonnull Class<?> type) {
242         requireNonBlank(name, ERROR_NAME_BLANK);
243         requireNonNull(type, ERROR_TYPE_NULL);
244         Object value = classPropertyFetcher.getPropertyValue(name);
245         return classPropertyFetcher.returnOnlyIfInstanceOf(value, type);
246     }
