55      private final Logger log;

57      @Inject

59      @GuardedBy("lock")
60      private GriffonClass griffonClass;
61  
62      public AbstractGriffonArtifact() {
63          log = LoggerFactory.getLogger("griffon.app." + getArtifactType() + "." + getTypeClass().getName());
64      }

91      @Nonnull
92      @Override
93      @SuppressWarnings("ConstantConditions")
94      public GriffonClass getGriffonClass() {
95          synchronized (lock) {
96              if (griffonClass == null) {
97                  griffonClass = application.getArtifactManager().findGriffonClass(getTypeClass());
98              }
99              return griffonClass;
100         }
101     }

448     /**
449      * Try to resolve the message.
450      *
451      * @param key Key to lookup, such as 'log4j.appenders.console'
452      *
453      * @return The resolved message at the given key for the default locale
454      *
455      * @throws NoSuchMessageException if no message is found
456      * @since 2.4.0
457      */
458     @Nonnull
459     protected String msg(@Nonnull String key) throws NoSuchMessageException {
460         return getApplication().getMessageSource().getMessage(key);
461     }
462 
463     /**
464      * Try to resolve the message.
465      *
466      * @param key  Key to lookup, such as 'log4j.appenders.console'
467      * @param args Arguments that will be filled in for params within the message (params look like "{0}" within a
468      *             message, but this might differ between implementations), or null if none.
