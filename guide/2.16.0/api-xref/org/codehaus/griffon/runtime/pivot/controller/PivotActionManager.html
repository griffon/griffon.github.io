18  package org.codehaus.griffon.runtime.pivot.controller;
19  
20  import griffon.core.GriffonApplication;
21  import griffon.core.artifact.GriffonController;
22  import griffon.core.controller.Action;
23  import griffon.core.controller.ActionFactory;
24  import griffon.core.controller.ActionMetadataFactory;
25  import griffon.exceptions.InstanceMethodInvocationException;
26  import org.apache.pivot.wtk.Component;
27  import org.codehaus.griffon.runtime.core.controller.AbstractActionManager;
28  import org.slf4j.Logger;
29  import org.slf4j.LoggerFactory;
30  
31  import javax.annotation.Nonnull;
32  import javax.annotation.Nullable;
33  import javax.inject.Inject;
34  
35  import static griffon.util.GriffonClassUtils.EMPTY_ARGS;
36  import static griffon.util.GriffonClassUtils.invokeExactInstanceMethod;
37  import static griffon.util.GriffonNameUtils.getNaturalName;
38  import static griffon.util.GriffonNameUtils.isNotBlank;
39  import static griffon.util.TypeUtils.castToBoolean;
40  
41  /**
42   * @author Andres Almiray
43   * @since 2.0.0
44   */
45  public class PivotActionManager extends AbstractActionManager {
46      private static final Logger LOG = LoggerFactory.getLogger(PivotActionManager.class);
47  
48      private static final String EMPTY_STRING = "";
49      private static final String DOT = ".";
50      private static final String EQUALS = " = ";
51      private static final String KEY_NAME = "name";
52      private static final String KEY_DESCRIPTION = "description";
53      private static final String KEY_ENABLED = "enabled";
54  
55      @Inject
56      public PivotActionManager(@Nonnull GriffonApplication application, @Nonnull ActionFactory actionFactory, @Nonnull ActionMetadataFactory actionMetadataFactory) {
57          super(application, actionFactory, actionMetadataFactory);
58      }
59  
60      @Override
61      protected void doConfigureAction(@Nonnull final Action action, @Nonnull final GriffonController controller, @Nonnull final String normalizeNamed, @Nonnull final String keyPrefix) {
62          controller.getApplication().addPropertyChangeListener(GriffonApplication.PROPERTY_LOCALE, evt -> configureAction((PivotGriffonControllerAction) action, controller, normalizeNamed, keyPrefix));
63          configureAction((PivotGriffonControllerAction) action, controller, normalizeNamed, keyPrefix);
64      }
65  
66      protected void configureAction(@Nonnull PivotGriffonControllerAction action, @Nonnull GriffonController controller, @Nonnull String normalizeNamed, @Nonnull String keyPrefix) {
67          resolveName(action, controller, normalizeNamed, keyPrefix);
68          resolveDescription(action, controller, normalizeNamed, keyPrefix);
69          resolveEnabled(action, controller, normalizeNamed, keyPrefix);
70      }
71  
72      protected void resolveName(@Nonnull PivotGriffonControllerAction action, @Nonnull GriffonController controller, @Nonnull String normalizeNamed, @Nonnull String keyPrefix) {
73          String rsActionName = msg(keyPrefix, normalizeNamed, KEY_NAME, getNaturalName(normalizeNamed));
74          if (isNotBlank(rsActionName)) {
75              trace(keyPrefix + normalizeNamed, KEY_NAME, rsActionName);
76              action.setName(rsActionName);
77          }
78      }
79  
80      protected void resolveDescription(@Nonnull PivotGriffonControllerAction action, @Nonnull GriffonController controller, @Nonnull String normalizeNamed, @Nonnull String keyPrefix) {
81          String rsDescription = msg(keyPrefix, normalizeNamed, KEY_DESCRIPTION, EMPTY_STRING);
82          if (isNotBlank(rsDescription)) {
83              trace(keyPrefix + normalizeNamed, KEY_DESCRIPTION, rsDescription);
84              action.setDescription(rsDescription);
85          }
86      }
87  
88      protected void resolveEnabled(@Nonnull PivotGriffonControllerAction action, @Nonnull GriffonController controller, @Nonnull String normalizeNamed, @Nonnull String keyPrefix) {
89          String rsEnabled = msg(keyPrefix, normalizeNamed, KEY_ENABLED, "true");
90          if (isNotBlank(rsEnabled)) {
91              trace(keyPrefix + normalizeNamed, KEY_ENABLED, rsEnabled);
92              action.setEnabled(castToBoolean(rsEnabled));
93          }
94      }
95  
96      @Nullable
97      @Override
98      protected Object doInvokeAction(@Nonnull GriffonController controller, @Nonnull String actionName, @Nonnull Object[] updatedArgs) {
99          try {
100             return invokeExactInstanceMethod(controller, actionName, updatedArgs);
101         } catch (InstanceMethodInvocationException imie) {
102             if (imie.getCause() instanceof NoSuchMethodException) {
103                 // try again but this time remove the 1st arg if it's
104                 // descendant of org.apache.pivot.wtk.Component
105                 if (updatedArgs.length == 1 && updatedArgs[0] != null && Component.class.isAssignableFrom(updatedArgs[0].getClass())) {
106                     return invokeExactInstanceMethod(controller, actionName, EMPTY_ARGS);
107                 } else {
108                     throw imie;
109                 }
110             } else {
111                 throw imie;
112             }
113         }
114     }
115 
116     protected void trace(@Nonnull String actionKey, @Nonnull String key, @Nonnull String value) {
117         if (LOG.isTraceEnabled()) {
118             LOG.trace(actionKey + DOT + key + EQUALS + value);
119         }
120     }
121 }
