53  /**
54   * Handles generation of code for the {@code @Threading} annotation.
55   * <p/>
56   *
57   * @author Andres Almiray
58   * @since 2.0.0
59   */
60  @AnnotationHandlerFor(Threading.class)
61  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
62  public class ThreadingASTTransformation extends AbstractASTTransformation implements ThreadingAwareConstants, AnnotationHandler {
63      private static final ClassNode THREADING_CNODE = makeClassSafe(Threading.class);
64      private static final ClassNode GRIFFON_CONTROLLER_CNODE = makeClassSafe(GriffonController.class);
65  
66      /**
67       * Convenience method to see if an annotated node is {@code @Threading}.
68       *
69       * @param node the node to check
70       *
71       * @return true if the node is an event publisher
72       */
73      public static boolean hasThreadingAnnotation(AnnotatedNode node) {
74          for (AnnotationNode annotation : node.getAnnotations()) {
75              if (THREADING_CNODE.equals(annotation.getClassNode())) {
76                  return true;
77              }
78          }
79          return false;
80      }
81  
82      /**
83       * Handles the bulk of the processing, mostly delegating to other methods.
84       *
85       * @param nodes  the ast nodes
86       * @param source the source unit for the nodes
87       */
88      public void visit(ASTNode[] nodes, SourceUnit source) {
89          AnnotationNode annotation = (AnnotationNode) nodes[0];
90          AnnotatedNode node = (AnnotatedNode) nodes[1];
91  
92          Threading.Policy threadingPolicy = getThreadingPolicy(annotation);
93          if (threadingPolicy == Threading.Policy.SKIP) { return; }
94  
95          String threadingMethod = resolveThreadingMethod(threadingPolicy);
96  
97          if (node instanceof MethodNode) {
98              ClassNode declaringClass = node.getDeclaringClass();
99              if (declaringClass.implementsInterface(GRIFFON_CONTROLLER_CNODE)) {
100                 return;
101             }
102 
103             addThreadingHandlerIfNeeded(source, declaringClass);
104             handleMethodForInjection(declaringClass, (MethodNode) node, threadingMethod);
105         } else if (node instanceof ClassNode) {
106             ClassNode declaringClass = (ClassNode) node;
107             if (declaringClass.implementsInterface(GRIFFON_CONTROLLER_CNODE)) {
108                 return;
109             }
110 
111             addThreadingHandlerIfNeeded(source, declaringClass);
112             for (MethodNode methodNode : declaringClass.getAllDeclaredMethods()) {
113                 threadingPolicy = getThreadingPolicy(methodNode, threadingPolicy);
114                 threadingMethod = resolveThreadingMethod(threadingPolicy);
115                 handleMethodForInjection(declaringClass, methodNode, threadingMethod);
116             }
117         }
118     }
119 
120     private String resolveThreadingMethod(Threading.Policy threadingPolicy) {
121         String threadingMethod = METHOD_RUN_OUTSIDE_UI;
122         switch (threadingPolicy) {
123             case INSIDE_UITHREAD_SYNC:
124                 threadingMethod = METHOD_RUN_INSIDE_UI_SYNC;
125                 break;
126             case INSIDE_UITHREAD_ASYNC:
127                 threadingMethod = METHOD_RUN_INSIDE_UI_ASYNC;
128                 break;
129             case OUTSIDE_UITHREAD_ASYNC:
130                 threadingMethod = METHOD_RUN_OUTSIDE_UI_ASYNC;
131                 break;
132             case OUTSIDE_UITHREAD:
133             default:
134                 break;
135         }
136         return threadingMethod;
137     }
138 
139     public static Threading.Policy getThreadingPolicy(AnnotationNode annotation) {
140         PropertyExpression value = (PropertyExpression) annotation.getMember("value");
141         if (value == null) { return Threading.Policy.OUTSIDE_UITHREAD; }
142         return Threading.Policy.valueOf(value.getPropertyAsString());
143     }
144 
145     public static Threading.Policy getThreadingPolicy(MethodNode method, Threading.Policy defaultPolicy) {
146         List<AnnotationNode> annotations = method.getAnnotations(THREADING_CNODE);
147         if (!annotations.isEmpty()) {
148             return getThreadingPolicy(annotations.get(0));
149         }
150         return defaultPolicy;
151     }
152 
153     public static void handleMethodForInjection(ClassNode classNode, MethodNode method, String threadingMethod) {
154         MethodDescriptor md = methodDescriptorFor(method);
155         if (GriffonClassUtils.isPlainMethod(md) &&
156             !GriffonClassUtils.isEventHandler(md) &&
157             hasVoidOrDefAsReturnType(method)) {
158             wrapStatements(classNode, method, threadingMethod);
159         }
160     }
161 
162     private static boolean hasVoidOrDefAsReturnType(MethodNode method) {
163         Class<?> returnType = method.getReturnType().getTypeClass();
164         return returnType.equals(ClassHelper.DYNAMIC_TYPE.getTypeClass()) ||
165             returnType.equals(ClassHelper.VOID_TYPE.getTypeClass());
166     }
167 
168     private static MethodDescriptor methodDescriptorFor(MethodNode method) {
169         if (method == null) { return null; }
170         Parameter[] types = method.getParameters();
171         String[] parameterTypes = new String[types.length];
172         for (int i = 0; i < types.length; i++) {
173             parameterTypes[i] = newClass(types[i].getType()).getName();
174         }
175         return new MethodDescriptor(method.getName(), parameterTypes, method.getModifiers());
176     }
177 
178     private static void wrapStatements(ClassNode declaringClass, MethodNode method, String threadingMethod) {
179         Statement code = method.getCode();
180         Statement wrappedCode = wrapStatements(code, threadingMethod);
181         if (code != wrappedCode) {
182             method.setCode(wrappedCode);
183             for (Parameter param : method.getParameters()) {
184                 param.setClosureSharedVariable(true);
185             }
186         }
187     }
188 
189     private static Statement wrapStatements(Statement code, String threadingMethod) {
190         // TODO deal with non-block statements
191         if (!(code instanceof BlockStatement)) { return code; }
192 
193         BlockStatement codeBlock = (BlockStatement) code;
194         List<Statement> statements = codeBlock.getStatements();
195         if (statements.isEmpty()) { return code; }
196 
197         VariableScope variableScope = codeBlock.getVariableScope();
198         BlockStatement block = new BlockStatement();
199         VariableScope blockScope = variableScope.copy();
200         makeVariablesShared(blockScope);
201         block.setVariableScope(blockScope);
202         ClosureExpression closure = new ClosureExpression(Parameter.EMPTY_ARRAY, code);
203         VariableScope closureScope = variableScope.copy();
204         makeVariablesShared(closureScope);
205         closure.setVariableScope(closureScope);
206         block.addStatement(stmnt(new MethodCallExpression(THIS, threadingMethod, args(closure))));
207 
208         return block;
209     }
210 
211     private static void makeVariablesShared(VariableScope scope) {
212         for (Iterator<Variable> vars = scope.getReferencedLocalVariablesIterator(); vars.hasNext(); ) {
213             Variable var = vars.next();
214             var.setClosureSharedVariable(true);
215         }
216     }
