60  /**
61   * Handles generation of code for the {@code @EventPublisher} annotation.
62   * <p/>
63   * Generally, it adds (if needed) an EventRouter field and
64   * the needed add/removeEventListener methods to support the
65   * listeners.
66   * <p/>
67   *
68   * @author Andres Almiray
69   */
70  @AnnotationHandlerFor(griffon.transform.EventPublisher.class)
71  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
72  public class EventPublisherASTTransformation extends AbstractASTTransformation implements EventPublisherConstants, AnnotationHandler {
73      private static final Logger LOG = LoggerFactory.getLogger(EventPublisherASTTransformation.class);
74      private static final ClassNode EVENT_ROUTER_CNODE = makeClassSafe(EventRouter.class);
75      private static final ClassNode EVENT_PUBLISHER_CNODE = makeClassSafe(EventPublisher.class);
76      private static final ClassNode EVENT_PUBLISHER_FIELD_CNODE = makeClassSafe(DefaultEventPublisher.class);
77      private static final ClassNode EVENT_PUBLISHER_ANODE = makeClassSafe(griffon.transform.EventPublisher.class);
78  
79      /**
80       * Convenience method to see if an annotated node is {@code @EventPublisher}.
81       *
82       * @param node the node to check
83       *
84       * @return true if the node is an event publisher
85       */
86      public static boolean hasEventPublisherAnnotation(AnnotatedNode node) {
87          for (AnnotationNode annotation : node.getAnnotations()) {
88              if (EVENT_PUBLISHER_ANODE.equals(annotation.getClassNode())) {
89                  return true;
90              }
91          }
92          return false;
93      }
94  
95      /**
96       * Handles the bulk of the processing, mostly delegating to other methods.
97       *
98       * @param nodes  the ast nodes
99       * @param source the source unit for the nodes
100      */
101     public void visit(ASTNode[] nodes, SourceUnit source) {
102         checkNodesForAnnotationAndType(nodes[0], nodes[1]);
103         addEventPublisherIfNeeded(source, (AnnotationNode) nodes[0], (ClassNode) nodes[1]);
104     }
105 
106     public static void addEventPublisherIfNeeded(SourceUnit source, AnnotationNode annotationNode, ClassNode classNode) {
107         if (needsDelegate(classNode, source, METHODS, "EventPublisher", EVENT_PUBLISHER_TYPE)) {
108             LOG.debug("Injecting {} into {}", EVENT_PUBLISHER_TYPE, classNode.getName());
109             ConstantExpression value = (ConstantExpression) annotationNode.getMember("value");
110             String beanName = value != null ? value.getText() : null;
111             beanName = isBlank(beanName) ? null : beanName;
112             apply(classNode, beanName);
113         }
114     }
115 
116     /**
117      * Adds the necessary field and methods to support resource locating.
118      *
119      * @param declaringClass the class to which we add the support field and methods
120      */
121     public static void apply(@Nonnull ClassNode declaringClass, @Nullable String beanName) {
122         injectInterface(declaringClass, EVENT_PUBLISHER_CNODE);
123 
124         FieldNode epField = injectField(declaringClass,
125             EVENT_PUBLISHER_FIELD_NAME,
126             PRIVATE,
127             EVENT_PUBLISHER_FIELD_CNODE,
128             ctor(EVENT_PUBLISHER_FIELD_CNODE));
129 
130         Parameter erParam = param(EVENT_ROUTER_CNODE, EVENT_ROUTER_PROPERTY);
131         if (isNotBlank(beanName)) {
132             AnnotationNode namedAnnotation = new AnnotationNode(NAMED_TYPE);
133             namedAnnotation.addMember("value", new ConstantExpression(beanName));
134             erParam.addAnnotation(namedAnnotation);
135         }
136 
137         MethodNode setter = new MethodNode(
138             METHOD_SET_EVENT_ROUTER,
139             PRIVATE,
140             VOID_TYPE,
141             params(erParam),
142             NO_EXCEPTIONS,
143             stmnt(call(
144                 field(epField),
145                 METHOD_SET_EVENT_ROUTER,
146                 args(var(EVENT_ROUTER_PROPERTY)))));
147         setter.addAnnotation(new AnnotationNode(INJECT_TYPE));
148         injectMethod(declaringClass, setter);
149 
150         addDelegateMethods(declaringClass, EVENT_PUBLISHER_CNODE, field(epField));
151     }
