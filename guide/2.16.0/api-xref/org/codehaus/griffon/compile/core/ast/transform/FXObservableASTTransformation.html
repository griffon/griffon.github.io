180     /**
181      * Convenience method to see if an annotated node is {@code @FXObservable}.
182      *
183      * @param node the node to check
184      *
185      * @return true if the node is observable
186      */
187     public static boolean hasFXObservableAnnotation(AnnotatedNode node) {
188         for (AnnotationNode annotation : node.getAnnotations()) {
189             if (FXOBSERVABLE_CNODE.equals(annotation.getClassNode())) {
190                 return true;
191             }
192         }
193         return false;
194     }

196     /**
197      * This ASTTransformation method is called when the compiler encounters our annotation.
198      *
199      * @param nodes      An array of nodes.  Index 0 is the annotation that triggered the call, index 1
200      *                   is the annotated node.
201      * @param sourceUnit The SourceUnit describing the source code in which the annotation was placed.
202      */
203     @Override
204     public void visit(ASTNode[] nodes, SourceUnit sourceUnit) {
205         if (!(nodes[0] instanceof AnnotationNode) || !(nodes[1] instanceof AnnotatedNode)) {
206             throw new IllegalArgumentException("Internal error: wrong types: "
207                 + nodes[0].getClass().getName() + " / " + nodes[1].getClass().getName());
208         }
209 
210         AnnotationNode node = (AnnotationNode) nodes[0];
211         AnnotatedNode parent = (AnnotatedNode) nodes[1];
212         ClassNode declaringClass = parent.getDeclaringClass();
213 
214         if (parent instanceof FieldNode) {
215             int modifiers = ((FieldNode) parent).getModifiers();
216             if ((modifiers & Modifier.FINAL) != 0) {
217                 String msg = "@griffon.transform.FXObservable cannot annotate a final property.";
218                 generateSyntaxErrorMessage(sourceUnit, node, msg);
219             }
220             addJavaFXProperty(sourceUnit, node, declaringClass, (FieldNode) parent);
221         } else {
222             addJavaFXPropertyToClass(sourceUnit, node, (ClassNode) parent);
223         }
224     }

226     /**
227      * Adds a JavaFX property to the class in place of the original Groovy property.  A pair of synthetic
228      * getter/setter methods is generated to provide pseudo-access to the original property.
229      *
230      * @param source         The SourceUnit in which the annotation was found
231      * @param node           The node that was annotated
232      * @param declaringClass The class in which the annotation was found
233      * @param field          The field upon which the annotation was placed
234      */
235     private void addJavaFXProperty(SourceUnit source, AnnotationNode node, ClassNode declaringClass, FieldNode field) {
236         String fieldName = field.getName();
237         for (PropertyNode propertyNode : declaringClass.getProperties()) {
238             if (propertyNode.getName().equals(fieldName)) {
239                 if (field.isStatic()) {
240                     String message = "@griffon.transform.FXObservable cannot annotate a static property.";
241                     generateSyntaxErrorMessage(source, node, message);
242                 } else {
243                     createPropertyGetterSetter(declaringClass, propertyNode);
244                 }
245                 return;
246             }
247         }
248 
249         String message = "@griffon.transform.FXObservable must be on a property, not a field. Try removing the private, " +
250             "protected, or public modifier.";
251         generateSyntaxErrorMessage(source, node, message);
252     }

254     /**
255      * Iterate through the properties of the class and convert each eligible property to a JavaFX property.
256      *
257      * @param source    The SourceUnit
258      * @param node      The AnnotationNode
259      * @param classNode The declaring class
260      */
261     private void addJavaFXPropertyToClass(SourceUnit source, AnnotationNode node, ClassNode classNode) {
262         for (PropertyNode propertyNode : classNode.getProperties()) {
263             FieldNode field = propertyNode.getField();
264             // look to see if per-field handlers will catch this one...
265             if (hasFXObservableAnnotation(field)
266                 || ((field.getModifiers() & Modifier.FINAL) != 0)
267                 || field.isStatic()) {
268                 // explicitly labeled properties are already handled,
269                 // don't transform final properties
270                 // don't transform static properties
271                 continue;
272             }
273             createPropertyGetterSetter(classNode, propertyNode);
274         }
275     }

277     /**
278      * Creates the JavaFX property and three methods for accessing the property and a pair of
279      * getter/setter methods for accessing the original (now synthetic) Groovy property.  For
280      * example, if the original property was "String firstName" then these three methods would
281      * be generated:
282      * <p>
283      * public String getFirstName()
284      * public void setFirstName(String value)
285      * public StringProperty firstNameProperty()
286      *
287      * @param classNode    The declaring class in which the property will appear
288      * @param originalProp The original Groovy property
289      */
290     private void createPropertyGetterSetter(ClassNode classNode, PropertyNode originalProp) {
291         Expression initExp = originalProp.getInitialExpression();
292         originalProp.getField().setInitialValueExpression(null);
293 
294         PropertyNode fxProperty = createFXProperty(originalProp);
295 
296         List<AnnotationNode> methodAnnotations = new ArrayList<>();
297         List<AnnotationNode> fieldAnnotations = new ArrayList<>();
298         for (AnnotationNode annotation : originalProp.getField().getAnnotations()) {
299             if (FXOBSERVABLE_CNODE.equals(annotation.getClassNode())) { continue; }
300             Class annotationClass = annotation.getClassNode().getTypeClass();
301             Target target = (Target) annotationClass.getAnnotation(Target.class);
302             if (isTargetAllowed(target, ElementType.METHOD)) {
303                 methodAnnotations.add(annotation);
304             } else if (isTargetAllowed(target, ElementType.FIELD)) {
305                 fieldAnnotations.add(annotation);
306             }
307         }
308 
309         String getterName = "get" + MetaClassHelper.capitalize(originalProp.getName());
310         if (classNode.getMethods(getterName).isEmpty()) {
311             Statement getterBlock = createGetterStatement(createFXProperty(originalProp));
312             createGetterMethod(classNode, originalProp, getterName, getterBlock, methodAnnotations);
313             methodAnnotations = null;
314         } else {
315             wrapGetterMethod(classNode, originalProp.getName(), methodAnnotations);
316             methodAnnotations = null;
317         }
318 
319         String setterName = "set" + MetaClassHelper.capitalize(originalProp.getName());
320         if (classNode.getMethods(setterName).isEmpty()) {
321             Statement setterBlock = createSetterStatement(createFXProperty(originalProp));
322             createSetterMethod(classNode, originalProp, setterName, setterBlock, methodAnnotations);
323         } else {
324             wrapSetterMethod(classNode, originalProp.getName(), methodAnnotations);
325         }
326 
327         // We want the actual name of the field to be different from the getter (Prop vs Property) so
328         // that the getter takes precedence when we say this.somethingProperty.
329         FieldNode fxFieldShortName = createFieldNodeCopy(originalProp.getName() + "Prop", null, fxProperty.getField());
330         createPropertyAccessor(classNode, createFXProperty(originalProp), fxFieldShortName, initExp);
331 
332         classNode.removeField(originalProp.getName());
333         classNode.addField(fxFieldShortName);
334         fxFieldShortName.addAnnotations(fieldAnnotations);
335     }

337     private boolean isTargetAllowed(Target target, ElementType elementType) {
338         if (target == null) {
339             return false;
340         }
341         for (ElementType et : target.value()) {
342             if (et == elementType) {
343                 return true;
344             }
345         }
346         return false;
347     }

349     /**
350      * Creates a new PropertyNode for the JavaFX property based on the original property.  The new property
351      * will have "Property" appended to its name and its type will be one of the *Property types in JavaFX.
352      *
353      * @param orig The original property
354      *
355      * @return A new PropertyNode for the JavaFX property
356      */
357     private PropertyNode createFXProperty(PropertyNode orig) {
358         ClassNode origType = orig.getType();
359         ClassNode newType = PROPERTY_TYPE_MAP.get(origType);
360 
361         // For the ObjectProperty, we need to add the generic type to it.
362         if (newType == null) {
363             if (isTypeCompatible(ClassHelper.LIST_TYPE, origType) || isTypeCompatible(OBSERVABLE_LIST_CNODE, origType)) {
364                 newType = makeClassSafe(SIMPLE_LIST_PROPERTY_CNODE);
365                 GenericsType[] genericTypes = origType.getGenericsTypes();
366                 newType.setGenericsTypes(genericTypes);
367             } else if (isTypeCompatible(ClassHelper.MAP_TYPE, origType) || isTypeCompatible(OBSERVABLE_MAP_CNODE, origType)) {
368                 newType = makeClassSafe(SIMPLE_MAP_PROPERTY_CNODE);
369                 GenericsType[] genericTypes = origType.getGenericsTypes();
370                 newType.setGenericsTypes(genericTypes);
371             } else if (isTypeCompatible(SET_TYPE, origType) || isTypeCompatible(OBSERVABLE_SET_CNODE, origType)) {
372                 newType = makeClassSafe(SIMPLE_SET_PROPERTY_CNODE);
373                 GenericsType[] genericTypes = origType.getGenericsTypes();
374                 newType.setGenericsTypes(genericTypes);
375             } else { // Object Type
376                 newType = makeClassSafe(OBJECT_PROPERTY_CNODE);
377                 ClassNode genericType = origType;
378                 if (genericType.isPrimaryClassNode()) {
379                     genericType = ClassHelper.getWrapper(genericType);
380                 }
381                 newType.setGenericsTypes(new GenericsType[]{new GenericsType(genericType)});
382             }
383         }
384 
385         FieldNode fieldNode = createFieldNodeCopy(orig.getName() + "Property", newType, orig.getField());
386         return new PropertyNode(fieldNode, orig.getModifiers(), orig.getGetterBlock(), orig.getSetterBlock());
387     }

389     private boolean isTypeCompatible(ClassNode base, ClassNode target) {
390         return target.equals(base) || target.implementsInterface(base) || target.declaresInterface(base);
391     }

393     /**
394      * Creates a setter method and adds it to the declaring class.  The setter has the form:
395      * <p>
396      * void <setter>(<type> fieldName)
397      *
398      * @param declaringClass The class to which the method is added
399      * @param propertyNode   The property node being accessed by this setter
400      * @param setterName     The name of the setter method
401      * @param setterBlock    The code body of the method
402      */
403     protected void createSetterMethod(ClassNode declaringClass, PropertyNode propertyNode, String setterName,
404                                       Statement setterBlock, List<AnnotationNode> annotations) {
405         Parameter[] setterParameterTypes = {new Parameter(propertyNode.getType(), "value")};
406         int mod = propertyNode.getModifiers() | Opcodes.ACC_FINAL;
407 
408         MethodNode setter = new MethodNode(setterName, mod, ClassHelper.VOID_TYPE, setterParameterTypes,
409             ClassNode.EMPTY_ARRAY, setterBlock);
410         if (annotations != null) { setter.addAnnotations(annotations); }
411         setter.setSynthetic(true);
412         declaringClass.addMethod(setter);
413     }

464     /**
465      * Creates the body of a property access method that returns the JavaFX *Property instance.  If
466      * the original property was "String firstName" then the generated code would be:
467      * <p>
468      * if (firstNameProperty == null) {
469      * firstNameProperty = new javafx.beans.property.StringProperty()
470      * }
471      * return firstNameProperty
472      *
473      * @param classNode        The declaring class to which the JavaFX property will be added
474      * @param fxProperty       The new JavaFX property
475      * @param fxFieldShortName
476      * @param initExp          The initializer expression from the original Groovy property declaration
477      */
478     private void createPropertyAccessor(ClassNode classNode, PropertyNode fxProperty, FieldNode fxFieldShortName,
479                                         Expression initExp) {
480         FieldExpression fieldExpression = new FieldExpression(fxFieldShortName);
481 
482         ArgumentListExpression ctorArgs = initExp == null ?
483             ArgumentListExpression.EMPTY_ARGUMENTS :
484             new ArgumentListExpression(initExp);
485 
486         BlockStatement block = new BlockStatement();
487         ClassNode fxType = fxProperty.getType();
488         ClassNode implNode = PROPERTY_IMPL_MAP.get(fxType);
489         if (implNode == null) {
490             if (fxType.getTypeClass() == SIMPLE_LIST_PROPERTY_CNODE.getTypeClass()) {
491                 if (initExp != null && initExp instanceof ListExpression ||
492                     (initExp instanceof CastExpression &&
493                         (((CastExpression) initExp).getType().equals(LIST_TYPE) ||
494                             ((CastExpression) initExp).getType().declaresInterface(LIST_TYPE))) ||
495                     (initExp instanceof ConstructorCallExpression &&
496                         (((ConstructorCallExpression) initExp).getType().equals(LIST_TYPE) ||
497                             ((ConstructorCallExpression) initExp).getType().declaresInterface(LIST_TYPE)))
498                     ) {
499                     ctorArgs = new ArgumentListExpression(
500                         new MethodCallExpression(
501                             new ClassExpression(FXCOLLECTIONS_CNODE),
502                             "observableList",
503                             ctorArgs)
504                     );
505                 }
506                 implNode = fxType;
507             } else if (fxType.getTypeClass() == SIMPLE_MAP_PROPERTY_CNODE.getTypeClass()) {
508                 if (initExp != null && initExp instanceof MapExpression ||
509                     (initExp instanceof CastExpression &&
510                         (((CastExpression) initExp).getType().equals(MAP_TYPE) ||
511                             ((CastExpression) initExp).getType().declaresInterface(MAP_TYPE))) ||
512                     (initExp instanceof ConstructorCallExpression &&
513                         (((ConstructorCallExpression) initExp).getType().equals(MAP_TYPE) ||
514                             ((ConstructorCallExpression) initExp).getType().declaresInterface(MAP_TYPE)))
515                     ) {
516                     ctorArgs = new ArgumentListExpression(
517                         new MethodCallExpression(
518                             new ClassExpression(FXCOLLECTIONS_CNODE),
519                             "observableMap",
520                             ctorArgs)
521                     );
522                 }
523                 implNode = fxType;
524             } else if (fxType.getTypeClass() == SIMPLE_SET_PROPERTY_CNODE.getTypeClass()) {
525                 if (initExp != null && (initExp instanceof CastExpression &&
526                     (((CastExpression) initExp).getType().equals(SET_TYPE) ||
527                         ((CastExpression) initExp).getType().declaresInterface(SET_TYPE))) ||
528                     (initExp instanceof ConstructorCallExpression &&
529                         (((ConstructorCallExpression) initExp).getType().equals(SET_TYPE) ||
530                             ((ConstructorCallExpression) initExp).getType().declaresInterface(SET_TYPE)))
531                     ) {
532                     ctorArgs = new ArgumentListExpression(
533                         new MethodCallExpression(
534                             new ClassExpression(FXCOLLECTIONS_CNODE),
535                             "observableSet",
536                             ctorArgs)
537                     );
538                 }
539                 implNode = fxType;
540             } else {
541                 implNode = makeClassSafe(SIMPLE_OBJECT_PROPERTY_CNODE);
542                 GenericsType[] origGenerics = fxProperty.getType().getGenericsTypes();
543                 //List<GenericsType> copyGenericTypes = new ArrayList<GenericsType>();
544                 //for()
545                 implNode.setGenericsTypes(origGenerics);
546             }
547         }
548         Expression initExpression = new ConstructorCallExpression(implNode, ctorArgs);
549 
550         IfStatement ifStmt = new IfStatement(
551             new BooleanExpression(
552                 new BinaryExpression(
553                     fieldExpression,
554                     Token.newSymbol(Types.COMPARE_EQUAL, 0, 0),
555                     ConstantExpression.NULL
556                 )
557             ),
558             new ExpressionStatement(
559                 new BinaryExpression(
560                     fieldExpression,
561                     Token.newSymbol(Types.EQUAL, 0, 0),
562                     initExpression
563                 )
564             ),
565             EmptyStatement.INSTANCE
566         );
567         block.addStatement(ifStmt);
568         block.addStatement(new ReturnStatement(fieldExpression));
569 
570         String getterName = getFXPropertyGetterName(fxProperty);
571         MethodNode accessor = new MethodNode(getterName, fxProperty.getModifiers(), fxProperty.getType(),
572             Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, block);
573         accessor.setSynthetic(true);
574         classNode.addMethod(accessor);
575 
576         // Create the xxxxProperty() method that merely calls getXxxxProperty()
577         block = new BlockStatement();
578 
579         VariableExpression thisExpression = VariableExpression.THIS_EXPRESSION;
580         ArgumentListExpression emptyArguments = ArgumentListExpression.EMPTY_ARGUMENTS;
581 
582         MethodCallExpression getProperty = new MethodCallExpression(thisExpression, getterName, emptyArguments);
583         block.addStatement(new ReturnStatement(getProperty));
584 
585         String javaFXPropertyFunction = fxProperty.getName();
586 
587         accessor = new MethodNode(javaFXPropertyFunction, fxProperty.getModifiers(), fxProperty.getType(),
588             Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, block);
589         accessor.setSynthetic(true);
590         classNode.addMethod(accessor);
591 
592         // Create the xxxx() method that merely calls getXxxxProperty()
593         block = new BlockStatement();
594 
595         thisExpression = VariableExpression.THIS_EXPRESSION;
596         emptyArguments = ArgumentListExpression.EMPTY_ARGUMENTS;
597 
598         getProperty = new MethodCallExpression(thisExpression, getterName, emptyArguments);
599         block.addStatement(new ReturnStatement(getProperty));
600         javaFXPropertyFunction = fxProperty.getName().replace("Property", "");
601 
602         accessor = new MethodNode(javaFXPropertyFunction, fxProperty.getModifiers(), fxProperty.getType(),
603             Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, block);
604         accessor.setSynthetic(true);
605         classNode.addMethod(accessor);
606     }

608     /**
609      * Creates the body of a setter method for the original property that is actually backed by a
610      * JavaFX *Property instance:
611      * <p>
612      * Object $property = this.someProperty()
613      * $property.setValue(value)
614      *
615      * @param fxProperty The original Groovy property that we're creating a setter for.
616      *
617      * @return A Statement that is the body of the new setter.
618      */
619     protected Statement createSetterStatement(PropertyNode fxProperty) {
620         String fxPropertyGetter = getFXPropertyGetterName(fxProperty);
621         VariableExpression thisExpression = VariableExpression.THIS_EXPRESSION;
622         ArgumentListExpression emptyArgs = ArgumentListExpression.EMPTY_ARGUMENTS;
623 
624         MethodCallExpression getProperty = new MethodCallExpression(thisExpression, fxPropertyGetter, emptyArgs);
625 
626         ArgumentListExpression valueArg = new ArgumentListExpression(new Expression[]{new VariableExpression("value")});
627         MethodCallExpression setValue = new MethodCallExpression(getProperty, "setValue", valueArg);
628 
629         return new ExpressionStatement(setValue);
630     }

632     /**
633      * Creates the body of a getter method for the original property that is actually backed by a
634      * JavaFX *Property instance:
635      * <p>
636      * Object $property = this.someProperty()
637      * return $property.getValue()
638      *
639      * @param fxProperty The new JavaFX property.
640      *
641      * @return A Statement that is the body of the new getter.
642      */
643     protected Statement createGetterStatement(PropertyNode fxProperty) {
644         String fxPropertyGetter = getFXPropertyGetterName(fxProperty);
645         VariableExpression thisExpression = VariableExpression.THIS_EXPRESSION;
646         ArgumentListExpression emptyArguments = ArgumentListExpression.EMPTY_ARGUMENTS;
647 
648         // We're relying on the *Property() method to provide the return value - is this still needed??
649         //        Expression defaultReturn = defaultReturnMap.get(originalProperty.getType());
650         //        if (defaultReturn == null)
651         //            defaultReturn = ConstantExpression.NULL;
652 
653         MethodCallExpression getProperty = new MethodCallExpression(thisExpression, fxPropertyGetter, emptyArguments);
654         MethodCallExpression getValue = new MethodCallExpression(getProperty, "getValue", emptyArguments);
655 
656         return new ReturnStatement(new ExpressionStatement(getValue));
657     }

661      * error message.
662      *
663      * @param sourceUnit The SourceUnit
664      * @param node       The node that was annotated
