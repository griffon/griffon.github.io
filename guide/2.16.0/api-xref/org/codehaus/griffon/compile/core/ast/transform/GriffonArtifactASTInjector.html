18  package org.codehaus.griffon.compile.core.ast.transform;
19  
20  import griffon.core.GriffonApplication;
21  import griffon.core.artifact.GriffonClass;
22  import org.codehaus.griffon.compile.core.BaseConstants;
23  import org.codehaus.groovy.ast.ClassNode;
24  import org.codehaus.groovy.ast.FieldNode;
25  import org.codehaus.groovy.ast.MethodNode;
26  import org.slf4j.Logger;
27  import org.slf4j.LoggerFactory;
28  
29  import javax.annotation.Nonnull;
30  
31  import static java.lang.reflect.Modifier.FINAL;
32  import static java.lang.reflect.Modifier.PRIVATE;
33  import static java.lang.reflect.Modifier.PUBLIC;
34  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.NO_ARGS;
35  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.NO_EXCEPTIONS;
36  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.NO_PARAMS;
37  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.THIS;
38  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.args;
39  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.call;
40  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.constx;
41  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.field;
42  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.injectField;
43  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.injectMethod;
44  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.returns;
45  import static org.codehaus.groovy.ast.ClassHelper.CLASS_Type;
46  
47  /**
48   * @author Andres Almiray
49   * @since 2.0.0
50   */
51  public class GriffonArtifactASTInjector extends AbstractASTInjector implements BaseConstants {
52      private static final ClassNode GRIFFON_APPLICATION_TYPE = makeClassSafe(GriffonApplication.class);
53      private static final ClassNode GRIFFON_CLASS_TYPE = makeClassSafe(GriffonClass.class);
54      private static final ClassNode LOGGER_TYPE = makeClassSafe(Logger.class);
55      private static final ClassNode LOGGER_FACTORY_TYPE = makeClassSafe(LoggerFactory.class);
56  
57      public void inject(@Nonnull ClassNode classNode, @Nonnull String artifactType) {
58          injectedField(classNode, GRIFFON_APPLICATION_TYPE, APPLICATION);
59  
60          // GriffonClass getGriffonClass()
61          injectMethod(classNode, new MethodNode(
62              "getGriffonClass",
63              PUBLIC,
64              GRIFFON_CLASS_TYPE,
65              NO_PARAMS,
66              NO_EXCEPTIONS,
67              returns(call(
68                  applicationProperty(classNode, "artifactManager"),
69                  "findGriffonClass",
70                  args(call(THIS, "getClass", NO_ARGS))))
71          ));
72          String loggerCategory = "griffon.app." + artifactType + "." + classNode.getName();
73          FieldNode loggerField = injectField(classNode,
74              "this$logger",
75              FINAL | PRIVATE,
76              LOGGER_TYPE,
77              call(
78                  LOGGER_FACTORY_TYPE,
79                  "getLogger",
80                  args(constx(loggerCategory)))
81          );
82  
83          // Class getTypeClass()
84          injectMethod(classNode, new MethodNode(
85              "getTypeClass",
86              PUBLIC,
87              makeClassSafe(CLASS_Type),
88              NO_PARAMS,
89              NO_EXCEPTIONS,
90              returns(call(ClassNode.THIS, "getClass", NO_ARGS))
91          ));
92  
93          // Logger getLog()
94          injectMethod(classNode, new MethodNode(
95              "getLog",
96              PUBLIC,
97              LOGGER_TYPE,
98              NO_PARAMS,
99              NO_EXCEPTIONS,
100             returns(field(loggerField))
101         ));
102 
103         ThreadingAwareASTTransformation.apply(classNode);
104         MVCAwareASTTransformation.apply(classNode);
105         ResourcesAwareASTTransformation.apply(classNode);
106     }
107 }
