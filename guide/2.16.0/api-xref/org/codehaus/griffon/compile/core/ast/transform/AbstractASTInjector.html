18  package org.codehaus.griffon.compile.core.ast.transform;
19  
20  import org.codehaus.griffon.compile.core.ast.GriffonASTUtils;
21  import org.codehaus.groovy.ast.AnnotationNode;
22  import org.codehaus.groovy.ast.ClassHelper;
23  import org.codehaus.groovy.ast.ClassNode;
24  import org.codehaus.groovy.ast.FieldNode;
25  import org.codehaus.groovy.ast.MethodNode;
26  import org.codehaus.groovy.ast.expr.ConstantExpression;
27  import org.codehaus.groovy.ast.expr.Expression;
28  import org.codehaus.groovy.ast.expr.FieldExpression;
29  
30  import javax.annotation.Nonnull;
31  import javax.annotation.Nullable;
32  import javax.inject.Inject;
33  import javax.inject.Named;
34  import java.lang.reflect.Modifier;
35  
36  import static griffon.util.GriffonNameUtils.getGetterName;
37  import static griffon.util.GriffonNameUtils.isNotBlank;
38  import static java.lang.reflect.Modifier.isPrivate;
39  import static java.util.Objects.requireNonNull;
40  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.NO_ARGS;
41  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.NO_PARAMS;
42  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.THIS;
43  import static org.codehaus.griffon.compile.core.ast.GriffonASTUtils.call;
44  import static org.codehaus.groovy.ast.expr.VariableExpression.THIS_EXPRESSION;
45  
46  /**
47   * @author Andres Almiray
48   * @since 2.0.0
49   */
50  public abstract class AbstractASTInjector implements ASTInjector {
51      public static final ClassNode NAMED_TYPE = makeClassSafe(Named.class);
52      private static final ClassNode THREAD_TYPE = ClassHelper.make(Thread.class).getPlainNodeReference();
53      private static final ClassNode INJECT_TYPE = makeClassSafe(Inject.class);
54      private static final String PROPERTY_APPLICATION = "application";
55      private static final String METHOD_GET_APPLICATION = "getApplication";
56  
57      @Nonnull
58      public static ClassNode makeClassSafe(@Nonnull ClassNode classNode) {
59          requireNonNull(classNode, "Argument 'classNode' must not be null");
60          return classNode.getPlainNodeReference();
61      }
62  
63      @Nonnull
64      public static ClassNode makeClassSafe(@Nonnull Class<?> klass) {
65          requireNonNull(klass, "Argument 'klass' must not be null");
66          return makeClassSafe(ClassHelper.make(klass));
67      }
68  
69      @Nonnull
70      public static Expression currentThread() {
71          return call(THREAD_TYPE, "currentThread", NO_ARGS);
72      }
73  
74      @Nonnull
75      public static Expression myClass() {
76          return call(THIS_EXPRESSION, "getClass", NO_ARGS);
77      }
78  
79      @Nonnull
80      public static Expression myClassLoader() {
81          return call(myClass(), "getClassLoader", NO_ARGS);
82      }
83  
84      @Nonnull
85      public static Expression applicationExpression(@Nonnull ClassNode classNode) {
86          FieldNode field = classNode.getDeclaredField(PROPERTY_APPLICATION);
87          if (field != null) {
88              return new FieldExpression(field);
89          }
90          field = classNode.getField(PROPERTY_APPLICATION);
91          if (field != null && !isPrivate(field.getModifiers())) {
92              return new FieldExpression(field);
93          }
94  
95          MethodNode method = classNode.getDeclaredMethod(METHOD_GET_APPLICATION, NO_PARAMS);
96          if (method != null) {
97              return call(THIS, METHOD_GET_APPLICATION, NO_ARGS);
98          }
99          method = classNode.getMethod(METHOD_GET_APPLICATION, NO_PARAMS);
100         if (method != null && !isPrivate(method.getModifiers())) {
101             return call(THIS, METHOD_GET_APPLICATION, NO_ARGS);
102         }
103         throw new IllegalStateException("Cannot resolve application field nor getApplication() method on class " + classNode.getName());
104     }
105 
106     @Nonnull
107     public static Expression applicationProperty(@Nonnull ClassNode classNode, @Nonnull String property) {
108         return call(applicationExpression(classNode), getGetterName(property), NO_ARGS);
109     }
110 
111     @Nonnull
112     public static FieldExpression injectedField(@Nonnull ClassNode owner, @Nonnull ClassNode type, @Nonnull String name) {
113         return injectedField(owner, type, name, null);
114     }
115 
116     @Nonnull
117     public static FieldExpression injectedField(@Nonnull ClassNode owner, @Nonnull ClassNode type, @Nonnull String name, @Nullable String qualifierName) {
118         FieldNode fieldNode = GriffonASTUtils.injectField(owner, name, Modifier.PRIVATE, type, null, false);
119         if (fieldNode.getAnnotations(INJECT_TYPE) == null) {
120             fieldNode.addAnnotation(new AnnotationNode(INJECT_TYPE));
121         }
122         if (isNotBlank(qualifierName) && fieldNode.getAnnotations(NAMED_TYPE) == null) {
123             AnnotationNode namedAnnotation = new AnnotationNode(NAMED_TYPE);
124             namedAnnotation.addMember("value", new ConstantExpression(qualifierName));
125             fieldNode.addAnnotation(namedAnnotation);
126         }
127         return new FieldExpression(fieldNode);
128     }
129 }
