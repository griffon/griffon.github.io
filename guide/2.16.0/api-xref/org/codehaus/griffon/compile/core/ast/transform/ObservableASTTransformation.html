75  /**
76   * Handles generation of code for the {@code @Observable} annotation.
77   *
78   * @author Andres Almiray
79   */
80  @AnnotationHandlerFor(griffon.transform.Observable.class)
81  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
82  public class ObservableASTTransformation extends AbstractASTTransformation implements ObservableConstants, AnnotationHandler {
83      private static final Logger LOG = LoggerFactory.getLogger(ObservableASTTransformation.class);
84      private static final ClassNode OBSERVABLE_CNODE = makeClassSafe(Observable.class);
85      private static final ClassNode OBSERVABLE_ANNOTATION_CNODE = makeClassSafe(griffon.transform.Observable.class);
86  
87      /**
88       * Convenience method to see if an annotated node is {@code @Observable}.
89       *
90       * @param node the node to check
91       * @return true if the node is an event publisher
92       */
93      public static boolean hasObservableAnnotation(AnnotatedNode node) {
94          for (AnnotationNode annotation : node.getAnnotations()) {
95              if (OBSERVABLE_ANNOTATION_CNODE.equals(annotation.getClassNode())) {
96                  return true;
97              }
98          }
99          return false;
100     }
101 
102     /**
103      * Handles the bulk of the processing, mostly delegating to other methods.
104      *
105      * @param nodes  the ast nodes
106      * @param source the source unit for the nodes
107      */
108     public void visit(ASTNode[] nodes, SourceUnit source) {
109         if (!(nodes[0] instanceof AnnotationNode) || !(nodes[1] instanceof AnnotatedNode)) {
110             throw new IllegalArgumentException("Internal error: wrong types: "
111                 + nodes[0].getClass().getName() + " / " + nodes[1].getClass().getName());
112         }
113         AnnotationNode node = (AnnotationNode) nodes[0];
114         AnnotatedNode parent = (AnnotatedNode) nodes[1];
115 
116         if (hasVetoableAnnotation(parent)) {
117             // VetoableASTTransformation will handle both @Observable and @Vetoable
118             return;
119         }
120 
121         ClassNode declaringClass = parent.getDeclaringClass();
122         if (parent instanceof FieldNode) {
123             if ((((FieldNode) parent).getModifiers() & Modifier.FINAL) != 0) {
124                 source.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(
125                     new SyntaxException("@griffon.transform.Observable cannot annotate a final property.",
126                         node.getLineNumber(), node.getColumnNumber(), node.getLastLineNumber(), node.getLastColumnNumber()),
127                     source));
128             }
129 
130             if (hasVetoableAnnotation(parent.getDeclaringClass())) {
131                 // VetoableASTTransformation will handle both @Observable and @Vetoable
132                 return;
133             }
134             addObservableIfNeeded(source, node, declaringClass, (FieldNode) parent);
135         } else if (parent instanceof ClassNode) {
136             addObservableIfNeeded(source, (ClassNode) parent);
137         }
138     }
139 
140     public static boolean needsObservableSupport(ClassNode classNode, SourceUnit source) {
141         return needsDelegate(classNode, source, OBSERVABLE_METHODS, "Observable", OBSERVABLE_TYPE);
142     }
143 
144     public static void addObservableIfNeeded(SourceUnit source, ClassNode classNode) {
145         if (needsObservableSupport(classNode, source)) {
146             LOG.debug("Injecting {} into {}", OBSERVABLE_TYPE, classNode.getName());
147             apply(classNode);
148         }
149 
150         for (PropertyNode propertyNode : classNode.getProperties()) {
151             FieldNode field = propertyNode.getField();
152             // look to see if per-field handlers will catch this one...
153             if (hasObservableAnnotation(field)
154                 || ((field.getModifiers() & Modifier.FINAL) != 0)
155                 || field.isStatic()
156                 || hasVetoableAnnotation(field)) {
157                 // explicitly labeled properties are already handled,
158                 // don't transform final properties
159                 // don't transform static properties
160                 // VetoableASTTransformation will handle both @Observable and @Vetoable
161                 continue;
162             }
163             createListenerSetter(classNode, propertyNode);
164         }
165     }
166 
167     public static void addObservableIfNeeded(SourceUnit source, AnnotationNode annotationNode, ClassNode classNode, FieldNode field) {
168         String fieldName = field.getName();
169         for (PropertyNode propertyNode : classNode.getProperties()) {
170             if (propertyNode.getName().equals(fieldName)) {
171                 if (field.isStatic()) {
172                     //noinspection ThrowableInstanceNeverThrown
173                     source.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(
174                         new SyntaxException("@griffon.transform.Observable cannot annotate a static property.",
175                             annotationNode.getLineNumber(), annotationNode.getColumnNumber(), annotationNode.getLastLineNumber(), annotationNode.getLastColumnNumber()),
176                         source));
177                 } else {
178                     if (needsObservableSupport(classNode, source)) {
179                         LOG.debug("Injecting {} into {}", OBSERVABLE_TYPE, classNode.getName());
180                         apply(classNode);
181                     }
182                     createListenerSetter(classNode, propertyNode);
183                 }
184                 return;
185             }
186         }
187         //noinspection ThrowableInstanceNeverThrown
188         source.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(
189             new SyntaxException("@griffon.transform.Observable must be on a property, not a field. Try removing the private, protected, or public modifier.",
190                 annotationNode.getLineNumber(), annotationNode.getColumnNumber(), annotationNode.getLastLineNumber(), annotationNode.getLastColumnNumber()),
191             source));
192     }
193 
194     private static void createListenerSetter(ClassNode classNode, PropertyNode propertyNode) {
195         String setterName = getSetterName(propertyNode.getName());
196         if (classNode.getMethods(setterName).isEmpty()) {
197             Expression fieldExpression = new FieldExpression(propertyNode.getField());
198             Statement setterBlock = createBindableStatement(propertyNode, fieldExpression);
199 
200             // create method void <setter>(<type> fieldName)
201             createSetterMethod(classNode, propertyNode, setterName, setterBlock);
202         } else {
203             wrapSetterMethod(classNode, propertyNode.getName());
204         }
205     }
206 
207     private static Statement createBindableStatement(PropertyNode propertyNode, Expression fieldExpression) {
208         // create statementBody
209         return stmnt(call(
210             THIS,
211             METHOD_FIRE_PROPERTY_CHANGE,
212             args(
213                 constx(propertyNode.getName()),
214                 fieldExpression,
215                 assign(fieldExpression, var(VALUE)))));
216     }
217 
218     private static void createSetterMethod(ClassNode declaringClass, PropertyNode propertyNode, String setterName, Statement setterBlock) {
219         MethodNode setter = new MethodNode(
220             setterName,
221             propertyNode.getModifiers(),
222             VOID_TYPE,
223             params(param(propertyNode.getType(), VALUE)),
224             NO_EXCEPTIONS,
225             setterBlock);
226         setter.setSynthetic(true);
227         // add it to the class
228         declaringClass.addMethod(setter);
229     }
230 
231     private static void wrapSetterMethod(ClassNode classNode, String propertyName) {
232         String getterName = getGetterName(propertyName);
233         MethodNode setter = classNode.getSetterMethod(getSetterName(propertyName));
234 
235         if (setter != null) {
236             // Get the existing code block
237             Statement code = setter.getCode();
238 
239             VariableExpression oldValue = new VariableExpression("$oldValue");
240             VariableExpression newValue = new VariableExpression("$newValue");
241             BlockStatement block = new BlockStatement();
242 
243             // create a local variable to hold the old value from the getter
244             block.addStatement(decls(oldValue, call(THIS, getterName, NO_ARGS)));
245 
246             // call the existing block, which will presumably set the value properly
247             block.addStatement(code);
248 
249             // get the new value to emit in the event
250             block.addStatement(decls(newValue, call(THIS, getterName, NO_ARGS)));
251 
252             // add the firePropertyChange method call
253             block.addStatement(stmnt(call(
254                 THIS,
255                 METHOD_FIRE_PROPERTY_CHANGE,
256                 args(constx(propertyName), oldValue, newValue))));
257 
258             // replace the existing code block with our new one
259             setter.setCode(block);
260         }
261     }

263     /**
264      * Adds the necessary field and methods to support resource locating.
265      *
266      * @param classNode the class to which we add the support field and methods
267      */
268     public static void apply(ClassNode classNode) {
269         injectInterface(classNode, OBSERVABLE_CNODE);
270 
271         ClassNode pcsClassNode = makeClassSafe(PropertyChangeSupport.class);
272         ClassNode pceClassNode = makeClassSafe(PropertyChangeEvent.class);
273 
274         // add field:
275         // protected final PropertyChangeSupport this$propertyChangeSupport = new java.beans.PropertyChangeSupport(this)
276         FieldNode pcsField = injectField(classNode,
277             PROPERTY_CHANGE_SUPPORT_FIELD_NAME,
278             FINAL | PROTECTED,
279             pcsClassNode,
280             ctor(pcsClassNode, args(GriffonASTUtils.THIS)));
281 
282         addDelegateMethods(classNode, OBSERVABLE_CNODE, new FieldExpression(pcsField));
283 
284         // add method:
285         // void firePropertyChange(String name, Object oldValue, Object newValue) {
286         //     this$propertyChangeSupport.firePropertyChange(name, oldValue, newValue)
287         //  }
288         injectMethod(classNode,
289             new MethodNode(
290                 METHOD_FIRE_PROPERTY_CHANGE,
291                 PROTECTED,
292                 VOID_TYPE,
293                 params(
294                     param(STRING_TYPE, NAME),
295                     param(makeClassSafe(OBJECT_TYPE), OLD_VALUE),
296                     param(OBJECT_TYPE, NEW_VALUE)),
297                 NO_EXCEPTIONS,
298                 stmnt(call(
299                     field(pcsField),
300                     METHOD_FIRE_PROPERTY_CHANGE,
301                     args(var(NAME), var(OLD_VALUE), var(NEW_VALUE))))));
302 
303         // add method:
304         // void firePropertyChange(PropertyChangeEvent event) {
305         //     this$propertyChangeSupport.firePropertyChange(event)
306         //  }
307         injectMethod(classNode,
308             new MethodNode(
309                 METHOD_FIRE_PROPERTY_CHANGE,
310                 PROTECTED,
311                 VOID_TYPE,
312                 params(param(pceClassNode, EVENT)),
313                 NO_EXCEPTIONS,
314                 stmnt(call(
315                     field(pcsField),
316                     METHOD_FIRE_PROPERTY_CHANGE,
317                     args(var(EVENT))))));
318     }
