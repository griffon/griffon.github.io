16 package griffon.transform;
17 
18 import java.lang.annotation.Documented;
19 import java.lang.annotation.ElementType;
20 import java.lang.annotation.Retention;
21 import java.lang.annotation.RetentionPolicy;
22 import java.lang.annotation.Target;
23 
24 /**
25  * <p>Annotates a class.</p>
26  * <p/>
27  * <p>When annotating a class it indicates that it will be able to locate
28  * resources using Griffon's standard resource loading mechanism.</p>
29  * <p/>
30  * The following methods will be added to classes annotated with &#064;ResourcesAware
31  * <ul>
32  * <li><code>public URL getResourceAsURL(java.lang.String)</code><br/>
33  * &nbsp;&nbsp;&nbsp;&nbsp;returns the resource URL if found</li>
34  * <li><code>public URL getResourceAsStream(java.lang.String)</code><br/>
35  * &nbsp;&nbsp;&nbsp;&nbsp;opens a stream to the resource if found</li>
36  * <li><code>public List&gt;URL&lt; getResources(java.lang.String)</code><br/>
37  * &nbsp;&nbsp;&nbsp;&nbsp;returns all matching resources for the specified name</li>
38  * <li><code>public ClassLoader classloader()</code><br/>
39  * &nbsp;&nbsp;&nbsp;&nbsp;returns the classloader used to locate resources</li>
40  * <p/>
41  * </ul>
42  * <p/>
43  *
44  * @author Andres Almiray
45  */
46 @Documented
47 @Retention(RetentionPolicy.SOURCE)
48 @Target(ElementType.TYPE)
49 public @interface ResourcesAware {
50 }
